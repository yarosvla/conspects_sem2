\section{Циклические подпространства}

Пусть $\dim V = n < \infty$ и $A \in \text{End } V$. Для любого вектора $v \in V$ введём понятие циклического подпространства.

\begin{Definition}{Циклическое подпространство}{}
Циклическим подпространством $C_v$ называется минимальное $A$-инвариантное подпространство, содержащее вектор $v$. Оно определяется как пересечение всех таких подпространств:
\[
C_v = \bigcap\{W \subseteq V \mid W \text{ — $A$-инвариантное подпространство и } v \in W\}.
\]
\end{Definition}

\subsubsection*{Свойство инвариантности}
Поскольку пересечение $A$-инвариантных подпространств снова $A$-инвариантно, подпространство $C_v$ также является $A$-инвариантным. Кроме того, если $W$ — любое $A$-инвариантное подпространство и $v \in W$, то $C_v \subseteq W$.

\subsection{Подстановка оператора в многочлен}
В теорию линейных операторов удобно ввести понятие многочлена от оператора. Пусть $f \in K[x]$ --- многочлен вида:
\[
f = a_0 + a_1 x + \dots + a_n x^n
\]
Тогда можно определить оператор $f(A) \in \text{End } V$ следующим образом:
\[
f(A) = a_0 \cdot E_V + a_1 A + a_2 A^2 + \dots + a_n A^n
\]
где под степенью понимается композиция: $A^k = \underbrace{A \circ A \circ \dots \circ A}_{k}$.

\subsubsection*{Гомоморфизм подстановки}
Легко видеть, что отображение $\mathcal{E} \colon K[x] \to \text{End } V$, заданное как $f \mapsto f(A)$, является \textbf{гомоморфизмом колец}, то есть:
\begin{enumerate}
    \item $\mathcal{E}(1) = E_V$ (единица переходит в тождественный оператор);
    \item $\mathcal{E}(f + g) = \mathcal{E}(f) + \mathcal{E}(g)$;
    \item $\mathcal{E}(f \cdot g) = \mathcal{E}(f) \cdot \mathcal{E}(g)$.
\end{enumerate}

\subsection{Аннулятор вектора и базис циклического подпространства}

Пусть $v \in V$. Рассмотрим множество многочленов, которые «обнуляют» данный вектор при подстановке в них оператора $A$.

\begin{Definition}{Аннулятор вектора}{}
Множество $I_v = \{ f \in K[x] \mid f(A)(v) = 0 \}$ называется аннулятором вектора $v$.
Заметим, что $I_v = \ker (\text{ev}_v)$, где $\text{ev}_v$ — отображение подстановки, вычисляющее значение в векторе $v$.
\end{Definition}

\begin{Lemma}{Свойства аннулятора}{}
$I_v$ является ненулевым идеалом в кольце $K[x]$.
\end{Lemma}

\subsubsection*{Доказательство}
\begin{enumerate}
    \item $f, g \in I_v \Rightarrow (f+g)(A)(v) = (f(A) + g(A))(v) = f(A)v + g(A)v = 0 + 0 = 0$. Значит, $f+g \in I_v$.
    \item $g \in I_v, f \in K[x] \Rightarrow (fg)(A)(v) = f(A)(g(A)(v)) = f(A)(0) = 0$. Значит, $fg \in I_v$.
    \item Покажем, что $I_v \neq \{0\}$. Рассмотрим систему векторов $v, Av, A^2 v, \dots, A^n v$. Так как $\dim V = n$, эти $n+1$ векторов линейно зависимы. Следовательно, существуют $\alpha_i$, не все равные нулю, такие что:
    \[
    \alpha_0 v + \alpha_1 Av + \dots + \alpha_n A^n v = 0
    \]
    Это означает, что многочлен $f(x) = \alpha_0 + \alpha_1 x + \dots + \alpha_n x^n$ принадлежит $I_v$ и $f \neq 0$.
\end{enumerate}

Так как $K[x]$ — область главных идеалов (ОГИ), то идеал $I_v$ порождается одним элементом: $I_v = (M_{v,A})$.

\begin{Definition}{Минимальный аннулятор вектора}{}
Многочлен $M_{v,A} = x^d + \beta_{d-1}x^{d-1} + \dots + \beta_0$ (нормированный многочлен минимальной степени из $I_v$) называется минимальным аннулятором вектора $v$.
\end{Definition}

\subsection{Базис циклического подпространства}

\begin{Proposition}{}{}
Пусть $\deg M_{v,A} = d$. Тогда система векторов $v, Av, A^2 v, \dots, A^{d-1} v$ является базисом циклического подпространства $C_v$.
\end{Proposition}

\subsection{Характеристический многочлен и теорема Гамильтона-Кэли}

Вычислим характеристический многочлен оператора, ограниченного на циклическое подпространство $C_v$. Матрица оператора $A|_{C_v}$ в базисе $v, Av, \dots, A^{d-1}v$ является сопровождающей матрицей для многочлена $M_{v,A}$.

\subsubsection*{Вычисление определителя}
Разложим определитель $\chi_{A|_{C_v}}(x) = \det(A|_{C_v} - xE)$ по последнему столбцу:
\begin{align*}
\chi_{A|_{C_v}}(x) &= \begin{vmatrix} 
-x & 0 & \dots & 0 & -\beta_0 \\ 
1 & -x & \dots & 0 & -\beta_1 \\ 
0 & 1 & -x & \dots & -\beta_2 \\ 
\vdots & \vdots & \ddots & \vdots & \vdots \\ 
0 & 0 & \dots & 1 & -\beta_{d-1}-x 
\end{vmatrix} = \\
&= (-1)^{d+1}(-\beta_0) \cdot 1 + (-1)^{d+2}(-\beta_1)(-x) + (-1)^{d+3}(-\beta_2)(-x)^2 + \dots + (-1)^{2d}(-\beta_{d-1}-x)(-x)^{d-1} = \\
&= (-1)^d \beta_0 + (-1)^{d-1} \beta_1 (-x) + (-1)^{d-2} \beta_2 (-x)^2 + \dots + (-1)^1 \beta_{d-1}(-x)^{d-1} + (-x)^d = \\
&= (-1)^d (\beta_0 + \beta_1 x + \beta_2 x^2 + \dots + \beta_{d-1} x^{d-1} + x^d)
\end{align*}

Заметим, что полученное выражение с точностью до знака совпадает с минимальным аннулятором:
\[ \chi_{A|_{C_v}} = \pm M_{v,A} \]

\begin{Theorem}{Гамильтона-Кэли}{}
Пусть $\dim V < \infty$ и $A \in \text{End } V$. Тогда $\chi_A(A) = 0$.
\end{Theorem}

\subsubsection*{Идея доказательства}
Для любого вектора $v \in V$:
\begin{enumerate}
    \item Характеристический многочлен оператора на инвариантном подпространстве делит характеристический многочлен всего оператора: $\chi_{A|_{C_v}} \mid \chi_A$.
    \item Следовательно, $\chi_A = \chi_{A|_{C_v}} \cdot g = \pm M_{v,A} \cdot g$ для некоторого $g \in K[x]$.
    \item Тогда $\chi_A(A)(v) = g(A) \cdot (\pm M_{v,A}(A)(v))$. Так как $M_{v,A}$ — аннулятор вектора $v$, то $M_{v,A}(A)(v) = 0$.
    \item Значит, $\chi_A(A)(v) = 0$ для любого $v$, откуда $\chi_A(A) = \mathbf{0}$.
\end{enumerate}

\subsection{Минимальный многочлен и разложение пространства}

Из теоремы Гамильтона-Кэли следует, что множество многочленов, аннулирующих оператор $A$, не пусто.

\begin{Definition}{Минимальный многочлен оператора}{}
Минимальным многочленом $\mathcal{M}_A$ называется нормированный многочлен минимальной степени такой, что $\mathcal{M}_A(A) = \mathbf{0}$. 
Он порождает идеал аннуляторов оператора:
\[
(\mathcal{M}_A) = \{ f \in K[x] \mid f(A) = 0 \}
\]
\end{Definition}

\subsubsection*{Свойства}
\begin{enumerate}
    \item Из теоремы Гамильтона-Кэли следует, что $\mathcal{M}_A \mid \chi_A$ (минимальный многочлен делит характеристический).
    \item Любой корень характеристического многочлена является корнем минимального (и наоборот, с точностью до кратности).
\end{enumerate}

\subsection{Теорема о разложении (Лемма о ядрах)}

Пусть $(fg)(A) = 0$ и многочлены $f$ и $g$ взаимно просты: $(f, g) = 1$.

\begin{Proposition}{}{}
В этих условиях пространство $V$ разлагается в прямую сумму $A$-инвариантных подпространств:
\[
V = V_1 \oplus V_2, \qquad V_1 = \ker f(A), \quad V_2 = \ker g(A).
\]
При этом $f(A)|_{V_1} = 0$ и $g(A)|_{V_2} = 0$.
\end{Proposition}

\subsubsection*{Доказательство}
Так как $(f, g) = 1$, по теореме Безу существуют такие $a, b \in K[x]$, что $f \cdot a + g \cdot b = 1$. 
Подставим оператор $A$: $f(A)a(A) + g(A)b(A) = E$.
Для любого $v \in V$:
\[
v = \underbrace{f(A)a(A)v}_{v_2} + \underbrace{g(A)b(A)v}_{v_1}
\]
Заметим, что $v_1 \in \ker f(A)$, так как $f(A)v_1 = f(A)g(A)b(A)v = (fg)(A)b(A)v = 0$. Аналогично $v_2 \in \ker g(A)$.
Следовательно, $V = \ker f(A) + \ker g(A)$. Прямая сумма следует из того, что если $v \in \ker f(A) \cap \ker g(A)$, то
\[
v = a(A)f(A)v + b(A)g(A)v = 0 + 0 = 0.
\]

\subsection{Обобщение и циклическая структура}

В общем случае, если $\mathcal{M}_A = p_1^{d_1} \dots p_t^{d_t}$, где $p_j$ — попарно различные неприводимые многочлены, то:
\[
V = W_1 \oplus W_2 \oplus \dots \oplus W_t, \quad W_j = \ker p_j^{d_j}(A)
\]
Каждое $W_j$ в свою очередь разлагается в прямую сумму циклических подпространств:
\[
W_j = W_{j_1} \oplus \dots \oplus W_{j_e}, \quad W_{j_i} = C_{v_{j_i}}
\]

\subsection{Жорданова форма и сопровождающая матрица}

В случае, когда неприводимый многочлен линейный ($p_j = x - \lambda_j$), структуру оператора можно свести к изучению случая $\lambda_j = 0$. 

В базисе $E = \{v, Av, \dots, A^{d-1}v\}$ матрица оператора на циклическом подпространстве имеет вид жордановой клетки:
\[
[A|_{C_{v_{j,i}}}]_E = \begin{pmatrix} 
0 & 1 & \dots & 0 \\ 
0 & 0 & \dots & 0 \\ 
\vdots & \vdots & \ddots & 1 \\ 
0 & 0 & \dots & 0 
\end{pmatrix} = J_{m_{j_i}}(0)
\]

В общем случае матрица оператора будет блочно-диагональной, где блоки $L_{j_k}$ соответствуют сопровождающим матрицам для многочленов $f_j$:
\[
L = \begin{pmatrix} 
L_{j_1} & & \\ 
& L_{j_2} & \\ 
& & \ddots & \\ 
& & & L_{j_k} 
\end{pmatrix}
\]

Для многочлена $f = x^n + a_{n-1}x^{n-1} + \dots + a_0$ сопровождающая матрица $L_f$ записывается как:
\[
L_f = \begin{pmatrix} 
0 & 0 & \dots & -a_0 \\ 
1 & 0 & \dots & -a_1 \\ 
\vdots & \vdots & \ddots & \vdots \\ 
0 & 0 & \dots & -a_{n-1} 
\end{pmatrix}
\]

\subsection{Теория групп. Простейшие свойства}

Начинаем изучение алгебраических структур с одним законом композиции.

\subsection{Основные свойства операций в группе}

\begin{Proposition}{Закон сокращения}{}
Пусть $G$ — группа. Тогда для любых элементов $g, g_1, g_2 \in G$ выполняются следующие свойства:
\begin{enumerate}
    \item $g_1 \cdot g = g_2 \cdot g \Rightarrow g_1 = g_2$ (правое сокращение)
    \item $g \cdot g_1 = g \cdot g_2 \Rightarrow g_1 = g_2$ (левое сокращение)
\end{enumerate}
\end{Proposition}

\begin{Proposition}{Обобщённая ассоциативность}{}
Результат умножения в выражении $g_1 g_2 \dots g_n$ не зависит от порядка выполнения операций (расстановки скобок).
\end{Proposition}

\subsubsection*{Формальная запись}
Произвольное произведение $g_1 g_2 \dots g_n$ всегда может быть приведено к «левонормированному» виду:
\[
g_1 g_2 \dots g_n = (\dots((g_1 \cdot g_2) \cdot g_3) \dots ) \cdot g_n
\]
Любой другой порядок вычислений (например, $(g_1 g_2)(g_3 g_4)$) в силу аксиомы ассоциативности даст тот же самый элемент группы.

\subsubsection*{Доказательство (индукция по $n$)}
\begin{itemize}
    \item \textbf{База:} $n=2$ — по аксиоме группы $g_1 g_2 = g_1 g_2$.
    \item \textbf{Переход:} $n-1 \to n$. 
    Рассмотрим произведение $n$ элементов, разбитое на две части: 
    \[ (g_1 \dots g_k) \cdot (g_{k+1} \dots g_n) \] 
    По предположению индукции (и.п.) каждая часть в скобках не зависит от расстановки внутренних скобок. Обозначим $a = (g_1 \dots g_k)$, $b = (g_{k+1} \dots g_{n-1})$ и $c = g_n$. 
    Тогда по аксиоме ассоциативности:
    \[ a \cdot (bc) = (ab) \cdot c = ((g_1 \dots g_k) \cdot (g_{k+1} \dots g_{n-1})) \cdot g_n \]
    Снова применяя и.п. к выражению в больших скобках, получаем левонормированное произведение:
    \[ (g_1 \dots g_{n-1}) \cdot g_n = \dots = g_1 \cdot g_2 \dots g_n \]
\end{itemize}

\subsection{Целая степень элемента группы}

Пусть $g \in G$ и $n \in \ZZ$.

\begin{Definition}{Степень элемента}{}
Степень элемента $g^n$ определяется следующим образом:
\[
g^n = \begin{cases} 
\underbrace{g \cdot g \cdot \dots \cdot g}_{n}, & n > 0 \\ 
e, & n = 0 \\ 
\underbrace{g^{-1} \cdot g^{-1} \cdot \dots \cdot g^{-1}}_{|n|}, & n < 0 
\end{cases}
\]
\end{Definition}

\begin{Proposition}{Свойства степеней}{}
Для любых $k, l \in \ZZ$:
\begin{enumerate}
    \item $g^k \cdot g^l = g^{k+l}$
    \item $(g^k)^l = g^{kl}$
\end{enumerate}
\end{Proposition}

\subsubsection*{Доказательство (случай $k > 0, l < 0, k+l > 0$)}
Пусть $l = -t$, где $t \in \NN$. Тогда:
\[
g^k \cdot g^l = g^k \cdot g^{-t} = \underbrace{g \cdot \dots \cdot g}_{k} \cdot \underbrace{g^{-1} \cdot \dots \cdot g^{-1}}_{t}
\]
Так как $k > t$, мы можем последовательно сократить $t$ пар элементов $g$ и $g^{-1}$:
\[
\underbrace{g \cdot \dots \cdot g}_{k-t} \cdot \underbrace{(g \cdot g^{-1}) \cdot \dots \cdot (g \cdot g^{-1})}_{t} = \underbrace{g \cdot \dots \cdot g}_{k-t} \cdot e = g^{k-t} = g^{k+l}
\]
Остальные случаи рассматриваются аналогично.

\subsection{Порядок элемента и циклические подгруппы}

\begin{Definition}{Порядок элемента}{}
Если $\exists n \in \NN$ такой, что $g^n = e$, то говорят, что элемент $g$ имеет \textbf{конечный порядок}. 
Порядком элемента $g$ называется число $\text{ord } g = \min \{n \in \NN \mid g^n = e\}$. 
В противном случае говорят, что $\text{ord } g = \infty$.
\end{Definition}

\begin{Proposition}{}{}
Пусть $\text{ord } g = d < \infty$. Тогда множество целых чисел, обнуляющих степень элемента, совпадает с идеалом $d\ZZ$:
\[ \{m \in \ZZ \mid g^m = e\} = d\ZZ \]
То есть $g^m = e \iff d \mid m$.
\end{Proposition}

\subsubsection*{Доказательство}
Разделим $m$ на $d$ с остатком: $m = dq + r$, где $0 \le r < d$.
Тогда:
\[ g^m = e \implies g^{dq+r} = e \implies (g^d)^q \cdot g^r = e \]
Так как $g^d = e$, получаем $e^q \cdot g^r = e$, откуда $g^r = e$. 
Поскольку $d$ — минимальная натуральная степень, дающая $e$, а $0 \le r < d$, то единственный возможный вариант: $r = 0$. Следовательно, $m = dq$, что и означало $d \mid m$.

\subsection{Подгруппа, порождённая элементом}

Для любого элемента $g \in G$ можно рассмотреть множество всех его целых степеней.

\begin{Definition}{Циклическая подгруппа}{}
Множество $\langle g \rangle = \{ g^n \mid n \in \ZZ \}$ называется циклической подгруппой, порождённой элементом $g$.
\end{Definition}

\subsubsection*{Доказательство подгрупповых аксиом}
Проверим, что $\langle g \rangle$ действительно является подгруппой в $G$:
\begin{enumerate}
    \item $e \in \langle g \rangle$, так как $e = g^0$.
    \item $h \in \langle g \rangle \implies h = g^n$. Тогда $h^{-1} = (g^n)^{-1} = g^{-n} \in \langle g \rangle$.
    \item $h_1, h_2 \in \langle g \rangle \implies h_1 = g^{n_1}, h_2 = g^{n_2}$. Тогда $h_1 h_2 = g^{n_1} g^{n_2} = g^{n_1+n_2} \in \langle g \rangle$.
\end{enumerate}
Таким образом, $\langle g \rangle < G$.

\begin{Proposition}{}{}
Пусть $g \in G$.
\begin{enumerate}
    \item Если $\text{ord } g = \infty$, то $|\langle g \rangle| = \infty$ и все степени $g^i \neq g^j$ при $i \neq j$.
	\item Если $\text{ord } g = n < \infty$, то $|\langle g \rangle| = n$ и $\langle g \rangle = \{g^0, g^1, \dots, g^{n-1}\}$.
\end{enumerate}
\end{Proposition}

    

\subsubsection*{Доказательство}
\begin{enumerate}
    \item Пусть $\text{ord } g = \infty$. Предположим, что $g^i = g^j$ при $i > j$. Тогда $g^i \cdot g^{-j} = g^j \cdot g^{-j} \Rightarrow g^{i-j} = e$. Так как $i-j > 0$, это противоречит условию $\text{ord } g = \infty$. Значит, все степени различны.
    
    \item Пусть $\text{ord } g = n < \infty$. 
    \begin{itemize}
        \item Для любого $m \in \ZZ$ представим $m = nq + r$, где $0 \le r < n$. Тогда $g^m = (g^n)^q \cdot g^r = e^q \cdot g^r = g^r$. Следовательно, $\langle g \rangle = \{g^0, g^1, \dots, g^{n-1}\}$.
        \item Покажем, что все эти элементы различны. Если $g^i = g^j$ для $0 \le j < i \le n-1$, то $g^{i-j} = e$. Но $0 < i-j \le n-1$, что противоречит тому, что $n$ — минимальная натуральная степень (порядок).
    \end{itemize}
\end{enumerate}
Т.о. $|\langle g \rangle| = \text{ord } g$.

\begin{Definition}{Циклическая группа}{}
Если существует такой элемент $g \in G$, что $\langle g \rangle = G$, то группа $G$ называется циклической, а элемент $g$ — её \textbf{образующим} (порождающим).
\end{Definition}

\subsubsection*{Таблица Кэли для циклической группы}
Для группы $\langle g \rangle = \{g^0, g^1, \dots, g^{n-1}\}$ таблица умножения имеет «сдвинутую» структуру:
\[
\begin{pmatrix} 
g^0 & g^1 & g^2 & \dots & g^{n-1} \\ 
g^1 & g^2 & g^3 & \dots & g^0 \\ 
g^2 & g^3 & g^4 & \dots & g^1 \\ 
\vdots & \vdots & \vdots & \ddots & \vdots \\ 
g^{n-1} & g^0 & g^1 & \dots & g^{n-2} 
\end{pmatrix}
\]

\subsubsection*{Примеры}
Основным примером циклической группы порядка $n$ является группа вычетов по модулю $n$ по сложению: $(\ZZ/n\ZZ, +)$. Здесь $\ZZ/n\ZZ = \langle \bar{1} \rangle$.

\subsection{Критерий цикличности конечной группы}

Пример: $\ZZ^* = \{ \pm 1 \}$ (группа обратимых элементов целых чисел по умножению).

\begin{Proposition}{}{}
Пусть порядок группы конечен: $|G| = n < \infty$. 
Тогда $G$ является циклической тогда и только тогда, когда в ней существует элемент порядка $n$.
\end{Proposition}

\subsubsection*{Доказательство}
\begin{itemize}
    \item[$\Rightarrow$)] Если $G$ циклическая, то по определению $G = \langle a \rangle$ для некоторого $a \in G$. Из свойств порядка следует, что $\text{ord } a = |G| = n$.
    \item[$\Leftarrow$)] Пусть $\exists g \in G$ такой, что $\text{ord } g = n$. Тогда порядок порождённой им циклической подгруппы $|\langle g \rangle| = n$. Так как $\langle g \rangle \subseteq G$ и их порядки (количество элементов) совпадают, то $\langle g \rangle = G$. Следовательно, $G$ — циклическая по определению.
\end{itemize}
