\section{Параллельные алгоритмы. Приложения: многопоточность на \texttt{CPU}, цифровая электроника, \texttt{FPGA}/\texttt{ПЛИС}, \texttt{GPU}}

\subsection{Вычислительные схемы}

\begin{Definition}{}{def:circuit}
	Вычислительная схема --- это ациклический ориентированный граф, состоящий из:
	\begin{enumerate}
		\item $n$ входных вершин, для которых $\operatorname{indeg}(v)=0$;
		\item $m$ выходных вершин, для которых $\operatorname{outdeg}(v)=0$;
		\item внутренних вершин, каждая из которых помечена элементарной операцией.
	\end{enumerate}
	В качестве элементарных операций могут выступать арифметические, булевы, а также любые другие операции, считающиеся простыми в данной модели.
	
	\medskip\noindent
	\textbf{Размер схемы} --- число её внутренних вершин.
	
	\medskip\noindent
	\textbf{Глубина схемы} --- длина максимального ориентированного пути от входа к выходу.
\end{Definition}

\begin{Remark}{}{rem:circuit-time}
	Глубина схемы интерпретируется как параллельное время вычисления: все вершины одного \emph{слоя} (то есть вершины, у которых все входящие данные уже готовы) могут вычисляться одновременно.
\end{Remark}

Алгоритм в модели вычислительных схем --- это семейство схем $\{C_n\}_{n\ge 1}$, где схема $C_n$ решает задачу на входах размера $n$.

\begin{Example}{}{ex:circuit-simple}
	Простейший пример: схема вычисления $(x_1+x_2)\cdot(x_3 - x_4)$ имеет четыре входа, два слоя и размер~$3$.
\end{Example}

\subsection{Схема быстрого преобразования Фурье}

Пусть $n=2^m$, $\omega$ --- примитивный корень степени $n$ из единицы, а входной вектор разбит на чётную и нечётную части.
Запустив FFT рекурсивно на каждой половине, получаем векторы $y^{(0)}$ и $y^{(1)}$ --- результаты DFT для чётных и нечётных компонент соответственно.
Тогда объединение (слой <<бабочек>>) имеет вид:
\[
y_k = y_k^{(0)} + \omega^k\, y_k^{(1)},
\qquad
y_{k+n/2} = y_k^{(0)} - \omega^k\, y_k^{(1)},
\qquad k=0,1,\dots,\tfrac{n}{2}-1.
\]

Рекурсивная схема FFT состоит из двух блоков $\mathrm{FFT}_{n/2}$, к выходам которых добавляется один слой из $n/2$ бабочек: $k$-я бабочка умножает правый вход на $\omega^k$, а затем складывает и вычитает.

\subsubsection*{Сложность}

Для схемы FFT выполняются оценки:
\[
\text{глубина} = O(\log n),
\qquad
\text{размер} = O(n\log n).
\]
Глубина удовлетворяет рекуррентности $T(n)=T(n/2)+O(1)$, откуда $T(n)=O(\log n)$.
Размер удовлетворяет $S(n)=2S(n/2)+O(n)$, откуда по мастер-теореме $S(n)=O(n\log n)$.

\begin{Proposition}{}{rem:fft-exercise}
	Пусть $y = F_{\omega} a$, где
	\[
	F_\omega = (\omega^{jk})_{j,k=0}^{n-1},
	\]
	а $\omega$ --- примитивный корень степени $n$ из единицы.
	Тогда
	\[
	F_\omega^2 = nP,
	\]
	где $P$ --- перестановка $P(a_0,a_1,\dots,a_{n-1}) = (a_0,a_{n-1},a_{n-2},\dots,a_1).$

\end{Proposition}

\begin{proof}
	Для любого $a=(a_0,\dots,a_{n-1})^T$ и любого $i=0,\dots,n-1$ имеем
	\[
	(F_\omega^2 a)_i
	=
	\sum_{j=0}^{n-1}\omega^{ij}\sum_{k=0}^{n-1}\omega^{jk}a_k
	=
	\sum_{k=0}^{n-1} a_k \sum_{j=0}^{n-1}\omega^{j(i+k)}.
	\]
	Но
	\[
	\sum_{j=0}^{n-1}\omega^{jt}
	=
	\begin{cases}
		n, & t \equiv 0 \pmod n,\\
		0, & t \not\equiv 0 \pmod n.
	\end{cases}
	\]
	Следовательно,
	\[
	(F_\omega^2 a)_i = n\,a_{-i \bmod n}.
	\]
	Итак, $\omega^2 a = n\,(a_0,a_{n-1},a_{n-2},\dots,a_1)^T = nPa,$ то есть $F_\omega^2 = nP.$

	Отсюда, если $y=F_\omega a$ и $\tilde a = F_\omega y$, то
	\[
	\tilde a = F_\omega^2 a = nPa,
	\qquad
	a = \frac1n P\tilde a.
	\]
\end{proof}

	Итак, чтобы восстановить $a$ по $y$, достаточно:
	\begin{enumerate}
		\item вычислить $\tilde a = F_\omega y$;
		\item обратить порядок координат $\tilde a_1,\dots,\tilde a_{n-1}$, оставив $\tilde a_0$ на месте;
		\item разделить все координаты на $n$.
	\end{enumerate}


\subsection{Сортирующие сети}

\begin{Definition}{}{def:sorting-network}
	\emph{Сортирующая сеть} --- это схема, которая по входу из $n$ чисел $x_1,\dots,x_n$ выдаёт числа $y_1,\dots,y_n$, упорядоченные по неубыванию, и использует в качестве внутренних элементов только \emph{компараторы}.
\end{Definition}

\begin{Definition}{}{def:comparator}
	\emph{Компаратор} с входами $u,v$ и выходами $u',v'$ вычисляет
	\[
	u' = \min\{u,v\},
	\qquad
	v' = \max\{u,v\}.
	\]
\end{Definition}

\subsubsection*{Нижние оценки}
\begin{itemize}
	\item \textbf{Размер} любой сортирующей сети не менее $\Omega(n\log n)$.
	Поскольку каждый компаратор порождает не более $2$ различных исходов, а число различных упорядочений $n$ элементов равно $n!$, то для различения всех входов нужно не менее $\log_2(n!)=\Omega(n\log n)$ компараторов.
	\item \textbf{Глубина} любой сортирующей сети не менее $\Omega(\log n)$.
	В схеме глубины $d$ значение одного выходного провода может зависеть не более чем от $2^d$ входных проводов.
	Так как максимальный элемент на выходе должен зависеть от всех $n$ входов, получаем $2^d \ge n$, откуда $d \ge \log_2 n$.
\end{itemize}

\subsection{Принцип нулей и единиц}

\begin{Theorem}{}{thm:zero-one}
	Сортирующая сеть корректна на всех вещественных входах тогда и только тогда, когда она корректна на всех входах из нулей и единиц.
\end{Theorem}

\begin{proof}
	Необходимость очевидна: $\{0,1\}$-входы являются частным случаем вещественных.
	
	Докажем достаточность.
	Предположим, что сеть корректно сортирует все $\{0,1\}$-входы, но не сортирует некоторый вещественный вход $x=(x_1,\dots,x_n)$.
	Тогда на выходе получается последовательность $y=(y_1,\dots,y_n)$, для которой существуют индексы $i<j$ такие, что $y_i>y_j$.
	
	Выберем число $t$ так, что $y_j < t \le y_i$, и определим булевый вход
	\[
	\widetilde x_k =
	\begin{cases}
		0, & x_k < t,\\
		1, & x_k \ge t.
	\end{cases}
	\]
	
	% ===========================================================
	% Рис. 6. Принцип нулей и единиц
	% ===========================================================
	\begin{figure}[h]\centering
		\begin{tikzpicture}[>=Stealth, font=\small]
			% Левый блок: вещественные входы
			\begin{scope}[xshift=0cm]
				\node[above,font=\small] at (-0.7,0.3) {$x_k$};
				\node[above,font=\small] at ( 0.7,0.3) {$x_h$};
				\draw[thin] (-0.7, 0.15) -- (-0.7,-1.9);
				\draw[thin] ( 0.7, 0.15) -- ( 0.7,-1.9);
				\draw[thick] (-0.7,-0.9) -- (0.7,-0.9);
				\filldraw (-0.7,-0.9) circle (2.5pt);
				\filldraw ( 0.7,-0.9) circle (2.5pt);
				\node[below,font=\scriptsize] at (-0.9,-2.05) {$\min\{x_k,x_h\}$};
				\node[below,font=\scriptsize] at ( 0.9,-2.05) {$\max\{x_k,x_h\}$};
			\end{scope}
			
			% Стрелка с пороговой функцией
			\node[font=\small] at (2.6,-0.9)
			{$\xrightarrow{\quad [\,\cdot\ge t\,]\quad}$};
			
			% Правый блок: пороговые (0/1) входы
			\begin{scope}[xshift=5.2cm]
				\node[above,font=\small] at (-0.7,0.3) {$\widetilde{x}_k$};
				\node[above,font=\small] at ( 0.7,0.3) {$\widetilde{x}_h$};
				\draw[thin] (-0.7, 0.15) -- (-0.7,-1.9);
				\draw[thin] ( 0.7, 0.15) -- ( 0.7,-1.9);
				\draw[thick] (-0.7,-0.9) -- (0.7,-0.9);

			\end{scope}
			
		\end{tikzpicture}
		\caption{
			Применение пороговой функции $[\,\cdot\ge t\,]$ до компаратора
			даёт тот же результат, что применение после:
			$[\min\{x_k,x_h\}\ge t] = \min\{[\,x_k\ge t\,],\,[\,x_h\ge t\,]\}$
			и аналогично для $\max$.}
		\label{fig:zero-one}
	\end{figure}
	
	\textbf{Ключевое наблюдение.} Пороговая функция $[\cdot \ge t]$ коммутирует с каждым компаратором.
	Действительно, рассмотрим компаратор на входах $u,v$, вычисляющий $(\min\{u,v\},\max\{u,v\})$:
	\begin{itemize}
		\item Если $u,v < t$, то $\min$ и $\max$ тоже $<t$; после порогования: $(0,0)$.
		Пороговые значения $(\widetilde u,\widetilde v)=(0,0)$; компаратор на них даёт $(0,0)$. Совпадает.
		\item Если $u,v \ge t$, то $\min$ и $\max$ тоже $\ge t$; после порогования: $(1,1)$.
		$(\widetilde u,\widetilde v)=(1,1)$; компаратор на них даёт $(1,1)$. Совпадает.
		\item Если $u < t \le v$ (или наоборот), то $\min\{u,v\}<t$ и $\max\{u,v\}\ge t$; после порогования: $(0,1)$.
		$(\widetilde u,\widetilde v)=(0,1)$ (или $(1,0)$); компаратор на них даёт $(0,1)$. Совпадает.
	\end{itemize}
	Таким образом, если сеть переводит $x$ в $y$, то она переводит $\widetilde x$ в $\widetilde y$, где
	\[
	\widetilde y_k =
	\begin{cases}
		0, & y_k < t,\\
		1, & y_k \ge t.
	\end{cases}
	\]
	Из выбора $t$ следует, что $\widetilde y_i=1$ и $\widetilde y_j=0$, то есть выход $\widetilde y$ не отсортирован.
	Получено противоречие с корректностью сети на $\{0,1\}$-входах.
\end{proof}

\subsection{Битонные последовательности}

\begin{Definition}{}{def:bitonic}
	Последовательность $z_1,\dots,z_m$ называется \emph{битонной}, если она является циклическим сдвигом последовательности, которая сначала неубывает, а затем невозрастает.
\end{Definition}

\begin{Example}{}{ex:bitonic-01}
	В случае $\{0,1\}$-последовательностей битонная последовательность имеет вид
	\[
	\underbrace{0\cdots 0}_{i}\,\underbrace{1\cdots 1}_{j}\,\underbrace{0\cdots 0}_{k}
	\qquad\text{или}\qquad
	\underbrace{1\cdots 1}_{i}\,\underbrace{0\cdots 0}_{j}\,\underbrace{1\cdots 1}_{k},
	\]
	где $i,j,k\ge 0$, то есть состоит не более чем из трёх чередующихся блоков.
\end{Example}

\subsection{Битонное слияние}

Пусть $m=2^t$. Для битонной последовательности $z_1,\dots,z_m$ определим первый слой компараторов:
\[
z_k^{\min} = \min\{z_k,\, z_{k+m/2}\},
\qquad
z_k^{\max} = \max\{z_k,\, z_{k+m/2}\},
\qquad k=1,\dots,\tfrac{m}{2}.
\]

\begin{Lemma}{}{bitonic-split}
	Пусть $z_1,\dots,z_m$ --- битонная последовательность.
	Тогда после одного слоя компараторов:
	\begin{enumerate}
		\item последовательность $(z_1^{\min},\dots,z_{m/2}^{\min})$ битонна;
		\item последовательность $(z_1^{\max},\dots,z_{m/2}^{\max})$ битонна;
		\item каждый элемент первой последовательности не превосходит каждого элемента второй.
	\end{enumerate}
	Более точно, в случае $\{0,1\}$-последовательностей: либо $z^{\min}=(0,\dots,0)$ и $z^{\max}$ битонна, либо $z^{\min}$ битонна и $z^{\max}=(1,\dots,1)$.
\end{Lemma}

Это позволяет рекурсивно продолжить сортировку: отсортировать обе половины независимо и объединить результаты.

\begin{algorithm}[H]
	\caption{\textsc{bi\_merge}$(z_1,\dots,z_m)$}
	\begin{algorithmic}[1]
		\Require Битонная последовательность $z_1,\dots,z_m$, где $m=2^t$ для некоторого $t \ge 0$
		\Ensure Последовательность $z_1,\dots,z_m$, отсортированная по неубыванию
		
		\If{$m = 1$}
		\State \Return $(z_1)$
		\EndIf
		
		\For{$k = 1,\dots,\frac{m}{2}$}
		\State $z_k^{\min} = \min(z_k, z_{k+m/2})$
		\State $z_k^{\max} = \max(z_k, z_{k+m/2})$
		\EndFor
		\Comment{один слой компараторов}
		
		\State $L = \textsc{bi\_merge}(z_1^{\min},\dots,z_{m/2}^{\min})$
		\Comment{рекурсивно обрабатываем меньшие элементы}
		
		\State $R = \textsc{bi\_merge}(z_1^{\max},\dots,z_{m/2}^{\max})$
		\Comment{рекурсивно обрабатываем большие элементы}
		
		\State \Return $L \mathbin\Vert R$
		\Comment{конкатенация двух половин}
	\end{algorithmic}
\end{algorithm}

\begin{Lemma}{}{lem:bi-merge-depth}
	Глубина сети \texttt{bi\_merge} на входе длины $m$ равна $O(\log m)$.
\end{Lemma}

\begin{proof}
	На каждом рекурсивном уровне добавляется один слой компараторов, после чего задача распадается на две независимые подзадачи длины $m/2$.
	Для глубины $T(m)$:
	\[
	T(m)=T\!\left(\tfrac{m}{2}\right)+O(1),
	\qquad T(m)=O(\log m).
	\]
\end{proof}

\begin{Lemma}{}{lem:bi-merge-size}
	Размер сети \texttt{bi\_merge} на входе длины $m$ равен $O(m\log m)$.
\end{Lemma}

\begin{proof}
	На верхнем уровне используется $m/2$ компараторов, после чего возникают две подзадачи длины $m/2$:
	\[
	S(m)=2S\!\left(\tfrac{m}{2}\right)+O(m),
	\qquad S(m)=O(m\log m).
	\]
\end{proof}

\subsection{Битонная сортировка}

Предполагается, что $n=2^t$; если это не так, вход дополняется значением $+\infty$ до ближайшей степени двойки.

\begin{algorithm}[H]
	\caption{\textsc{bitonic\_sort}$(x_1,\dots,x_n)$}
	\begin{algorithmic}[1]
		\Require Последовательность $x_1,\dots,x_n$, где $n = 2^t$ для некоторого $t \ge 0$
		\Ensure Последовательность из тех же элементов, отсортированная по неубыванию
		
		\If{$n = 1$}
		\State \Return $(x_1)$
		\EndIf
		
		\State $(a_1,\dots,a_{n/2}) = \textsc{bitonic\_sort}(x_1,\dots,x_{n/2})$
		\Comment{левая половина сортируется по неубыванию}
		
		\State $(b_1,\dots,b_{n/2}) = \textsc{bitonic\_sort}(x_{n/2+1},\dots,x_n)$
		\Comment{правая половина сортируется по неубыванию}
		
		\State \Return \textsc{bi\_merge}$(a_1,\dots,a_{n/2},\, b_{n/2},\dots,b_1)$
		\Comment{получаем битонную последовательность и сливаем её}
	\end{algorithmic}
\end{algorithm}

\noindent
Конкатенация $(a_1,\dots,a_{n/2},b_{n/2},\dots,b_1)$ является битонной: первая половина возрастает, вторая убывает.

\begin{Remark}{}{}
	Корректность битонной сортировки на всех вещественных входах вытекает из теоремы о принципе нулей и единиц.
	Для $\{0,1\}$-входа достаточно убедиться, что все $n$ одинаковых элементов и блоки $0\cdots0\,1\cdots1$ обрабатываются правильно, что проверяется по лемме~\ref{lem:bitonic-split}.
\end{Remark}

\begin{Theorem}{}{thm:bitonic-sort-complexity}
	Битонная сортировка имеет глубину $O(\log^2 n)$ и размер $O(n\log^2 n)$.
\end{Theorem}

\begin{proof}
	Обозначим через $D(n)$ глубину схемы, а через $S(n)$ --- её размер.
	
	\textbf{Глубина.}
	После рекурсивной сортировки половин (глубина $D(n/2)$) выполняется \texttt{bi\_merge} глубины $O(\log n)$:
	\[
	D(n)=D\!\left(\tfrac{n}{2}\right)+O(\log n).
	\]
	Раскрывая рекуррентность:
	\[
	D(n)=\sum_{k=0}^{\log n - 1} O\!\left(\log\tfrac{n}{2^k}\right)
	= O\!\left(\sum_{k=0}^{\log n - 1}(\log n - k)\right)
	= O(\log^2 n).
	\]
	
	\textbf{Размер.}
	Шаг \texttt{bi\_merge} имеет размер $O(n\log n)$, поэтому:
	\[
	S(n)=2S\!\left(\tfrac{n}{2}\right)+O(n\log n),
	\qquad S(n)=O(n\log^2 n).
	\]
\end{proof}

\subsection{Нечётно-чётная сортировка (Batcher)}

Альтернативный способ построения сортирующей сети с той же асимптотикой --- нечётно-чётная сортировка Батчера.
Ключевая операция --- \emph{нечётно-чётное слияние} двух уже отсортированных последовательностей.

\begin{algorithm}[H]
	\caption{\textsc{odd\_even\_merge}$(a_1,\dots,a_m;\, b_1,\dots,b_m)$}
	\begin{algorithmic}[1]
		\Require Две отсортированные по неубыванию последовательности длины $m$, где $m=2^t$
		\Ensure Последовательность длины $2m$, полученная слиянием входов и отсортированная по неубыванию
		
		\If{$m = 1$}
		\State $u = \min(a_1,b_1)$
		\State $v = \max(a_1,b_1)$
		\State \Return $(u,v)$
		\EndIf
		
		\State $(c_1,\dots,c_m) = \textsc{odd\_even\_merge}(a_1,a_3,\dots,a_{m-1};\, b_1,b_3,\dots,b_{m-1})$
		\Comment{нечётные позиций}
		
		\State $(d_1,\dots,d_m) = \textsc{odd\_even\_merge}(a_2,a_4,\dots,a_m;\, b_2,b_4,\dots,b_m)$
		\Comment{чётные позиций}
		
		\State $r_1 = c_1$
		\For{$i = 1,\dots,m-1$}
		\State $r_{2i} = \min(c_{i+1}, d_i)$
		\State $r_{2i+1} = \max(c_{i+1}, d_i)$
		\EndFor
		\State $r_{2m} = d_m$
		\Comment{финальный слой из $m-1$ компараторов}
		
		\State \Return $(r_1,\dots,r_{2m})$
	\end{algorithmic}
\end{algorithm}


\begin{Lemma}{}{lem:odd-even-merge-correct}
	Процедура \textsc{odd\_even\_merge} корректно сливает две отсортированные последовательности длины $m$.
\end{Lemma}

\begin{proof}
	По принципу нулей и единиц рассмотрим вход из $\{0,1\}$. Докажем корректность индукцией по $m$.
	
	\textbf{База:} $m=1$ очевидна.
	
	\textbf{Шаг:} Пусть $m > 1$. По предположению индукции $c$ и $d$ отсортированы. Пусть $z_a, z_b$ — количество нулей во входах. Тогда в $c$ и $d$ их количество равно:
	\[ z_c = \lceil z_a/2 \rceil + \lceil z_b/2 \rceil, \quad z_d = \lfloor z_a/2 \rfloor + \lfloor z_b/2 \rfloor \]
	Так как $0 \le \lceil x/2 \rceil - \lfloor x/2 \rfloor \le 1$, разность $z_c - z_d$ может принимать значения $\{0, 1, 2\}$.
	
	Рассмотрим последовательность $c_1, d_1, c_2, d_2, \dots, c_m, d_m$. 
	Если $z_c - z_d \in \{0, 1\}$, она уже отсортирована. Если $z_c - z_d = 2$, то $c$ содержит на два нуля больше, чем $d$. Это создает единственную инверсию $(d_i=1, c_{i+1}=0)$ на границе раздела нулей и единиц. 
	
	Финальный слой компараторов $(d_i, c_{i+1})$ для всех $i$ устраняет эту возможную инверсию, делая последовательность полностью отсортированной.
\end{proof}

\begin{algorithm}[H]
	\caption{\textsc{odd\_even\_sort}$(x_1,\dots,x_n)$}
	\begin{algorithmic}[1]
		\Require Последовательность $x_1,\dots,x_n$, где $n=2^t$ для некоторого $t \ge 0$
		\Ensure Последовательность из тех же элементов, отсортированная по неубыванию
		
		\If{$n = 1$}
		\State \Return $(x_1)$
		\EndIf
		
		\State $(a_1,\dots,a_{n/2}) = \textsc{odd\_even\_sort}(x_1,\dots,x_{n/2})$
		\Comment{сортировка левой половины}
		
		\State $(b_1,\dots,b_{n/2}) = \textsc{odd\_even\_sort}(x_{n/2+1},\dots,x_n)$
		\Comment{сортировка правой половины}
		
		\State $\Return \textsc{odd\_even\_merge}(a_1,\dots,a_{n/2};\, b_1,\dots,b_{n/2})$
	\end{algorithmic}
\end{algorithm}

\begin{Remark}{}{}
	Корректность \textsc{odd\_even\_sort} доказывается индукцией по $n$:
	сначала рекурсивно сортируются левая и правая половины, после чего они корректно сливаются процедурой \textsc{odd\_even\_merge}.
\end{Remark}
