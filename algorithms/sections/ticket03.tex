\section{Префиксные суммы и сумматор Ладнера–Фишера}

\subsection{Задача о префиксных суммах}

Построить вычислительную схему, которая принимает на вход $n$ чисел $x_0, \dots, x_{n-1}$ и (используя только бинарную операцию «$+$») выдаёт числа $y_0, \dots, y_{n-1}$, где
\[
y_i = \sum_{j=0}^{i} x_j, \qquad i = 0, \dots, n-1.
\]

\subsection{Схема Ладнера–Фишера}

Пусть $n$ чётно. Алгоритм состоит из трёх фаз.

\begin{enumerate}
	\item \textbf{Свёртка (upsweep).} Попарно складываем соседние входы:
	\[
	s_i = x_{2i} + x_{2i+1}, \qquad i = 0, \dots, \tfrac{n}{2} - 1.
	\]
	
	\item \textbf{Рекурсия.} Считаем префиксные суммы для массива $s$:
	\[
	p_i = \sum_{j=0}^{i} s_j.
	\]
	
	\item \textbf{Развёртка (downsweep).} Восстанавливаем выходы:
	\[
	y_{2i+1} = p_i, \qquad
	y_{2i} =
	\begin{cases}
		x_0, & i = 0, \\
		p_{i-1} + x_{2i}, & i \geq 1.
	\end{cases}
	\]
\end{enumerate}

\begin{Statement}{Оценки глубины и размера схемы Ладнера–Фишера}{}
	Обозначим через $T(n)$ глубину схемы (длину критического пути), а через $S(n)$ — её размер (число операций). Тогда при $n = 2^m$:
	\begin{align*}
		T(n) &= 2 + T\!\left(\tfrac{n}{2}\right), \quad T(1) = 0
		\quad \Longrightarrow \quad T(n) = 2\log n, \\
		S(n) &= (n-1) + S\!\left(\tfrac{n}{2}\right)
		\quad \Longrightarrow \quad S(n) = 2n - \log n - 2.
	\end{align*}
\end{Statement}

\subsection{Обобщение: операция \texttt{scan}}

\begin{Definition}{}{}
	Операция \textbf{\texttt{scan}} (префиксное накопление) для ассоциативной бинарной операции $\odot$ на множестве $S$ (то есть для \textbf{полугруппы} $(S, \odot)$) определяется как вычисление массива
	\[
	y_i = x_0 \odot x_1 \odot \cdots \odot x_i, \qquad i = 0, \dots, n-1.
	\]
\end{Definition}

Схема Ладнера–Фишера работает для любой ассоциативной операции. Примеры допустимых операций $\odot$:
\[
(\mathbb{R},\, +), \quad
(\mathbb{R},\, \min), \quad
(\mathbb{R},\, \max), \quad
(\mathbb{R}^{k \times k},\, \cdot) \text{ (умножение матриц)}.
\]

\subsection{Псевдокод \texttt{scan} (in-place, inclusive)}

Ниже представлен алгоритм для $n = 2^m$. Для фиксированного $d$ все операции во внутреннем цикле по $k$ выполняются \emph{параллельно}.

\begin{algorithm}[H]
	\caption{\texttt{scan} (in-place, inclusive)}
	\begin{algorithmic}[1]
		\Require массив $x[0..n-1]$, $n = 2^m$
		\Statex \textbf{Upsweep:} накапливаем суммы на концах блоков
		\For{$d = 0$ \textbf{to} $\log_2 n - 1$}
		\For{$k = 0$ \textbf{to} $n-1$ \textbf{step} $2^{d+1}$}
		\State $x[k + 2^{d+1} - 1] \mathrel{+}= x[k + 2^d - 1]$
		\EndFor
		\EndFor
		\Statex \textbf{Downsweep:} добавляем префикс предыдущего блока к левому подблоку
		\For{$d = \log_2 n - 1$ \textbf{downto} $0$}
		\For{$k = 0$ \textbf{to} $n-1$ \textbf{step} $2^{d+1}$}
		\If{$k > 0$}
		\State $x[k + 2^d - 1] \mathrel{+}= x[k - 1]$
		\EndIf
		\EndFor
		\EndFor
	\end{algorithmic}
\end{algorithm}

\subsection{Сведение к \texttt{scan}: линейная рекуррента}

\begin{Example}{Рекуррента вида $c_{k+1} = a_k + b_k c_k$}{}
	Даны числа $a_0, \dots, a_{n-1}$ и $b_0, \dots, b_{n-1}$. Требуется найти $c_0, \dots, c_n$ такие, что
	\[
	c_0 = 0, \qquad c_{k+1} = a_k + b_k\, c_k.
	\]
	
	Определим строчный вектор $C_k = (1,\ c_k)$. Тогда
	\[
	C_{k+1} = C_k X_k, \qquad
	X_k = \begin{pmatrix} 1 & a_k \\ 0 & b_k \end{pmatrix}.
	\]
	Отсюда
	\[
	C_1 = C_0 X_0, \quad
	C_2 = C_0 X_0 X_1, \quad
	C_3 = C_0 X_0 X_1 X_2, \quad \dots
	\]
	Таким образом, задача сводится к вычислению \textbf{префиксных произведений} матриц $X_0, X_1, \dots, X_{n-1}$. Поскольку умножение матриц ассоциативно, это делается через \texttt{scan} за $O(\log n)$ времени в параллельной модели.
\end{Example}

\begin{Remark}{}{}
	В примере с матрицами при перемножении используются \emph{две} операции над элементами:
	\[
	(A \cdot B)_{ij} = \bigoplus_k (A_{ik} \odot B_{kj}).
	\]
	Поэтому обобщение работает не для одной операции, а для \textbf{полукольца} $(S, \oplus, \odot)$, в котором:
	\begin{itemize}
		\item $\oplus$ ассоциативна (обычно коммутативна) и имеет нейтральный элемент $0$;
		\item $\odot$ ассоциативна и имеет нейтральный элемент $1$;
		\item $\odot$ дистрибутивна относительно $\oplus$:
		\[
		a \odot (b \oplus c) = (a \odot b) \oplus (a \odot c), \qquad
		(a \oplus b) \odot c = (a \odot c) \oplus (b \odot c);
		\]
		\item $0$ поглощает умножение: $0 \odot a = a \odot 0 = 0$.
	\end{itemize}
	Примеры полуколец $(S, \oplus, \odot)$:
	\[
	\bigl(\{0,1\},\ \vee,\ \wedge\bigr), \qquad
	\bigl(\{0,1\},\ \oplus,\ \wedge\bigr), \qquad
	\bigl(\mathbb{R} \cup \{+\infty\},\ \min,\ +\bigr).
	\]
\end{Remark}

\subsection{Сумматор Ладнера–Фишера}

\begin{Definition}{}{}
	\textbf{Сумматор} — это вычислительная схема над булевыми операциями $(\vee, \wedge, \oplus, \neg, \dots)$, которая принимает на вход биты $x_0, \dots, x_{n-1}$ и $y_0, \dots, y_{n-1}$ и выдаёт биты $z_0, \dots, z_n$ такие, что $z = x + y$, где
	\[
	x = \sum_{k=0}^{n-1} x_k 2^k, \qquad
	y = \sum_{k=0}^{n-1} y_k 2^k, \qquad
	z = \sum_{k=0}^{n} z_k 2^k.
	\]
\end{Definition}

\subsubsection*{Биты переноса и сведение к \texttt{scan}}

Обозначим через $c_k$ перенос \emph{в} разряд $k$ (по определению $c_0 = 0$). Тогда для $k = 0, \dots, n-1$ выполняется \textbf{формула переноса}:
\[
c_{k+1} = (x_k \wedge y_k) \vee (x_k \wedge c_k) \vee (y_k \wedge c_k).
\]
Введём обозначения $a_k := x_k \wedge y_k$ и $b_k := x_k \vee y_k$. Тогда формула переноса принимает компактный вид:
\[
c_{k+1} = a_k \vee (b_k \wedge c_k).
\]
Если все переносы $c_0, \dots, c_n$ известны, ответ восстанавливается по формулам:
\[
z_k = x_k \oplus y_k \oplus c_k \quad (k = 0, \dots, n-1), \qquad z_n = c_n.
\]

\textbf{Ключевая идея: сведение к \texttt{scan}}

	Рекуррента $c_{k+1} = a_k \vee (b_k \wedge c_k)$ является аффинным преобразованием. Положим
	\[
	C_k = (1,\ c_k), \qquad
	X_k = \begin{pmatrix} 1 & a_k \\ 0 & b_k \end{pmatrix},
	\]
	где операции понимаются над булевым полукольцом $(\{0,1\},\ \vee,\ \wedge)$. Тогда
	\[
	C_{k+1} = C_k X_k,
	\]
	и задача нахождения всех переносов сводится к вычислению префиксных произведений
	\[
	C_k = C_0 X_0 X_1 \cdots X_{k-1},
	\]
	что выполняется через \texttt{scan} за $O(\log n)$ времени в параллельной модели.


\subsection{Модель PRAM}

\begin{Definition}{}{}
	\textbf{PRAM} (Parallel Random Access Machine) — идеализированная модель параллельных вычислений:
	\begin{itemize}
		\item имеется общая глобальная память;
		\item число процессоров теоретически не ограничено; каждый процессор является RAM-машиной с собственным номером $p = 1, 2, \dots$;
		\item процессоры работают \textbf{синхронно}: за единицу времени каждый выполняет одну инструкцию.
	\end{itemize}
\end{Definition}

\subsubsection*{Разрешение конфликтов доступа к памяти}

Различают два режима для чтения и записи: \emph{concurrent} (одновременный доступ нескольких процессоров к одной ячейке разрешён) и \emph{exclusive} (одновременный доступ запрещён). Стандартные варианты модели:
\[
\text{CRCW PRAM}, \quad \text{CREW PRAM}, \quad \text{ERCW PRAM}, \quad \text{EREW PRAM}.
\]
В модели \textbf{CRCW} при одновременной записи нескольких процессоров в одну ячейку необходимо задать правило разрешения конфликта (например, \emph{arbitrary}: в ячейке остаётся одно из записанных значений).

\begin{Definition}{}{}
	\textbf{NC} — класс задач, которые решаются на PRAM за время $(\log n)^{O(1)}$ при использовании полиномиального числа процессоров $n^{O(1)}$, где $n$ — размер входа.
\end{Definition}

\subsection{\texttt{Scan} на $p$ процессорах в модели PRAM}

Пусть дан массив $A[0..n-1]$ и $p$ процессоров. Разобьём массив на блоки длины $m = n/p$ (считаем $p \mid n$).

\begin{enumerate}
	\item Каждый процессор $i$ вычисляет \textbf{локальный \texttt{scan}} на своём блоке $A[im..(i+1)m-1]$ последовательно за время $O(m) = O(n/p)$. Обозначим сумму $i$-го блока через $T_i$.
	
	\item Вычисляем \textbf{\texttt{scan} по массиву блок-сумм} $T_0, T_1, \dots, T_{p-1}$ за время $O(\log p)$. Пусть
	\[
	P_i = \sum_{j=0}^{i-1} T_j
	\]
	— сумма всех предшествующих блоков (смещение для блока $i$).
	
	\item Каждый процессор прибавляет смещение $P_i$ ко всем элементам своего блока за время $O(n/p)$.
\end{enumerate}

\begin{Statement}{Время работы \texttt{scan} на $p$ процессорах}{}
	Суммарное время составляет
	\[
	O\!\left(\frac{n}{p} + \log p\right).
	\]
	В частности, при $p = \Theta\!\left(\dfrac{n}{\log n}\right)$ получаем время $O(\log n)$.
\end{Statement}

\begin{Example}{Сортировка подсчётом (counting sort) в PRAM}{}
	Дан массив $a[0..n-1]$, где $a[k] \in \{0, \dots, U\}$ при $U = O(1)$. Используем $p$ процессоров с блоками длины $m = n/p$.
	
	\begin{enumerate}
		\item \textbf{Локальный подсчёт частот.} Процессор $i$ строит таблицу частот $c_i[0..U]$:
		\[
		c_i(x) = \#\{\, k \in [im,\, (i+1)m - 1] \mid a[k] = x \,\}.
		\]
		Время: $O(n/p)$.
		
		\item \textbf{Сборка и \texttt{scan}.} Формируем массив длины $(U+1)p$:
		\[
		C[xp + i] = c_i(x), \qquad x = 0, \dots, U,\ \ i = 0, \dots, p-1.
		\]
		Вычисляем \texttt{scan} по $C$ и получаем массив $P$.
		
		\item \textbf{Расстановка элементов.} Процессор $i$ знает, что элементы значения $x$ из его блока нужно поместить в позиции
		\[
		\bigl[\, P[xp + i] - c_i(x),\ P[xp + i) \,\bigr).
		\]
		Время: $O(n/p)$.
	\end{enumerate}
	
	\paragraph{Итоговое время.}
	Локальный подсчёт — $O(n/p)$; \texttt{scan} по массиву длины $(U+1)p$ — $O(\log p)$, так как $U = O(1)$. Итого:
	\[
	O\!\left(\frac{n}{p} + \log p\right).
	\]
\end{Example}
