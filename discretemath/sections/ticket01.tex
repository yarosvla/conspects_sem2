\section{Вершинная связность}

\begin{Definition}{Разделяющее множество}{}
	В графе $G$ множество вершин $R \subset V$ называется
	\emph{разделяющим} для множеств $X \subset V$ и $Y \subset V$
	(которые могут пересекаться), если в подграфе $G[V \setminus R]$
	не существует пути между вершинами из $X \setminus R$
	и вершинами из $Y \setminus R$.
\end{Definition}

\begin{Theorem}{Геринг}{}
	Пусть задан граф $G$ и два множества его вершин $X, Y$.
	Если каждое $XY$-разделяющее множество содержит хотя бы $k$ вершин,
	то существует не менее $k$ попарно непересекающихся по вершинам
	$XY$-путей (путей из вершины $X$ в вершину $Y$).
\end{Theorem}

\begin{Statement}{}{}
	Эквивалентная переформулировка: максимальное число попарно
	непересекающихся по вершинам $XY$-путей равно минимальному
	размеру $XY$-разделяющего множества.
\end{Statement}

\begin{proof}
	Доказательство ведём индукцией по числу вершин $|V|$.
	
	\medskip
	\noindent\textbf{База:} $|V| = k$.
	
	Заметим, что $R = X$ является $XY$-разделяющим множеством
	(после удаления $X$ множество $X \setminus R$ пусто),
	поэтому $|X| \geq k$. Аналогично $|Y| \geq k$.
	Поскольку $|V| = k$, заключаем $X = Y = V$.
	Тогда из каждой вершины в саму себя существует путь длины $0$;
	это ровно $k = |V|$ попарно непересекающихся путей. \checkmark
	
	\medskip
	\noindent\textbf{Переход:} $|V| > k$. Рассматриваем два случая.
	
	\begin{enumerate}
		\item \textbf{Существует «минимальное» разделяющее множество,
			отделяющее обе части.}
		
		Формально: $\exists\, R$ --- $XY$-разделяющее множество с
		$|R| = k$, $X \setminus R \neq \emptyset$,
		$Y \setminus R \neq \emptyset$.
		(В частности, $X \neq Y$.)
		
		\medskip
		Определим:
		\begin{itemize}
			\item $V_X$ --- множество вершин, достижимых из $X \setminus R$
			в $G \setminus R$;
			\item $V_Y$ --- множество вершин, достижимых из $Y \setminus R$
			в $G \setminus R$;
			\item $G_X = G[V_X \cup R]$, \quad $G_Y = G[V_Y \cup R]$.
		\end{itemize}
		
		Так как $R$ разделяет $X$ и $Y$, множества $V_X$ и $V_Y$
		не пересекаются. Следовательно, $|V(G_X)| < |V|$ и
		$|V(G_Y)| < |V|$, и к обоим подграфам применимо
		индукционное предположение.
		
		\begin{Statement}{}{}
			Любое $XR$-разделяющее множество в $G_X$ является
			также $XY$-разделяющим в $G$.
		\end{Statement}
		\begin{proof}
			Пусть $S$ --- $XR$-разделяющее множество в $G_X$, и
			предположим от противного, что в $G \setminus S$
			существует путь $p$ из $X$ в $Y$.
			
			Поскольку $R$ разделяет $X$ и $Y$ в $G$, путь $p$
			проходит через хотя бы одну вершину из $R$.
			Рассмотрим \emph{первую} такую вершину $r \in R$.
			Начальный отрезок пути $p$ от $X$ до $r$ не содержит
			вершин из $R \setminus \{r\}$, поэтому он целиком
			проходит в $G_X$.
			Этот отрезок соединяет $X$ с $r \in R$, не пересекая $S$
			--- противоречие с тем, что $S$ разделяет $X$ и $R$ в $G_X$.
		\end{proof}
		
		Отсюда: любое $XR$-разделяющее множество в $G_X$ имеет
		размер $\geq k$. По индукционному предположению в $G_X$
		найдётся $k$ непересекающихся $XR$-путей; аналогично в
		$G_Y$ --- $k$ непересекающихся $YR$-путей.
		Склеивая каждый $XR$-путь с соответствующим $YR$-путём
		через вершину $R$ и пользуясь тем, что $V_X \cap V_Y = \emptyset$,
		получаем $k$ попарно непересекающихся $XY$-путей.
		
		\item \textbf{Разделяющего множества из случая 1 не существует.}
		
		То есть любое $XY$-разделяющее множество $R$ с $|R| = k$
		поглощает $X$ или $Y$ целиком: $X \setminus R = \emptyset$
		или $Y \setminus R = \emptyset$.
		
		\begin{enumerate}
			\item \textbf{$X \cup Y \neq V$.}
			
			Выберем произвольную вершину $u \in V \setminus (X \cup Y)$. Заметим, что $u \notin X$
			и $u \notin Y$.
			
			\begin{Statement}{}{}
				В графе $G \setminus \{u\}$ не существует
				$XY$-разделяющего множества $S$ с $|S| = k - 1$.
			\end{Statement}
			
			\begin{proof}
				Предположим противное: в графе $G \setminus \{u\}$ существует
				$XY$-разделяющее множество $S$ такое, что $|S| = k - 1$.
				
				Так как $X$ и $Y$ сами являются $XY$-разделяющими множествами, то
				по определению числа $k$ имеем $k \le |X|$ и $k \le |Y|$.
				Следовательно, $|S| = k - 1 < |X|$ и $|S| = k - 1 < |Y|$.
				Поэтому множества $X \setminus S$ и $Y \setminus S$ непусты.
				
				Рассмотрим множество $S' = S \cup \{u\}$. Тогда $S'$ является
				$XY$-разделяющим множеством в графе $G$, причем
				$|S'| = |S| + 1 = k.$
				
				Кроме того, поскольку $u \notin X \cup Y$, имеем
				\[
				X \setminus S' = X \setminus S \neq \emptyset,
				\qquad
				Y \setminus S' = Y \setminus S \neq \emptyset.
				\]
				Получили $XY$-разделяющее множество в $G$ размера $k$, что
				противоречит предположению случая~2.
			\end{proof}
			
			Значит, в $G \setminus \{u\}$ все $XY$-разделяющие
			множества имеют размер $\geq k$. Граф $G \setminus \{u\}$
			содержит строго меньше вершин, чем $G$, поэтому по
			индукционному предположению в нём есть $k$ непересекающихся
			$XY$-путей, а следовательно, они есть и в $G$.
			
			\item \textbf{$X \cup Y = V$.}
			
			Заметим, что каждая вершина из $X \cap Y$ должна входить
			в любое $XY$-разделяющее множество.
			
			Рассмотрим граф $H = G[V \setminus (X \cap Y)]$.
			В нём вершины разбиваются на два класса:
			$X' = X \setminus Y$ и
			$Y' = Y \setminus X$,
			поэтому $H$ двудольный.
			
			Возьмём минимальное вершинное покрытие $C$ графа $H$
			и доложим к нему $X \cap Y$: множество $C \cup (X \cap Y)$
			является минимальным $XY$-разделяющим множеством в $G$.
			По теореме Кёнига:
			\[
			k
			= \bigl|C\bigr| + \bigl|X \cap Y\bigr|
			= \bigl|\text{макс. паросочетание в } H\bigr|
			+ \bigl|X \cap Y\bigr| 
			\]
			\[
			= \text{число непересекающихся } XY\text{-путей.}
			\]
		\end{enumerate}
	\end{enumerate}
\end{proof}

\begin{Theorem}{Менгер}{}
	Максимальное число попарно непересекающихся по внутренним
	вершинам путей между несмежными вершинами $v$ и $u$ равно
	минимальному размеру $vu$-разделяющего множества
	$S \subset V \setminus \{v, u\}$.
\end{Theorem}
\begin{proof}
	Пусть минимальное разделяющее множество имеет размер $k$.
	Заменим вершину $v$ на $k$ её клонов $v_1, \ldots, v_k$
	(каждый соединён с теми же соседями, что и $v$),
	и аналогично заменим $u$ на клоны $u_1, \ldots, u_k$.
	Пусть $X = \{v_1,\ldots,v_k\}$, $Y = \{u_1,\ldots,u_k\}$.
	По теореме Геринга максимальное число попарно непересекающихся
	$XY$-путей равно минимальному размеру $XY$-разделяющего
	множества, что соответствует $k$ путям в исходном графе.
\end{proof}

\begin{Definition}{$k$-связный граф}{}
	Граф $G$ называется \emph{$k$-связным}, если
	$|V(G)| > k$ и удаление любого множества $S \subset V(G)$
	с $|S| < k$ оставляет граф связным.
\end{Definition}

\begin{Definition}{}{}
	$\kappa_G(v, u)$ --- максимальное количество попарно
	непересекающихся по внутренним вершинам путей от $v$ до $u$
	в графе $G$.
\end{Definition}

\begin{Theorem}{}{}
	Граф $G$ является $k$-связным $\Leftrightarrow$
	$\kappa_G(v, u) \geq k$  $\forall \, v \neq u$.
\end{Theorem}
\begin{proof}
	$(\Rightarrow)$
	Пусть $G$ $k$-связен. Возьмём произвольные $v \neq u$.
	
	\begin{itemize}
		\item \textit{$v$ и $u$ несмежны.} Тогда $\kappa_G(v,u) \geq k$
		непосредственно из теоремы Менгера.
		
		\item \textit{$v$ и $u$ смежны.} Добавим в $G$ вспомогательные
		вершины $v'$ (соединённую с соседями $u$) и $u'$
		(соединённую с соседями $v$). Связность не понизится,
		поэтому по теореме Менгера существует $k$
		непересекающихся путей от $v'$ до $u'$.
		Заменим каждый такой путь на путь между $v$ и $u$:
		\begin{enumerate}
			\item путь содержит $v$ и $u$
			$\Rightarrow$ заменяем на ребро $\{v,u\}$;
			\item путь содержит только $v$
			$\Rightarrow$ берём отрезок от $u'$ до $v$,
			заменяя $u'$ на $u$;
			\item путь содержит только $u$
			$\Rightarrow$ берём отрезок от $u$ до $v'$,
			заменяя $v'$ на $v$;
			\item путь не содержит ни $v$, ни $u$
			$\Rightarrow$ заменяем $v'$ на $v$, $u'$ на $u$.
		\end{enumerate}
	\end{itemize}
	
	$(\Leftarrow)$
	Заметим, что $|V(G)| > k$: для любых $v \neq u$
	существуют $k$ внутренне непересекающихся путей между ними,
	из которых не более одного может быть ребром $\{v,u\}$, а каждый
	из остальных содержит хотя бы одну внутреннюю вершину; значит,
	в графе есть хотя бы $2+(k-1)=k+1$ вершина.
	
	Предположим от противного: существует $S$ с $|S| < k$ такое,
	что $G \setminus S$ несвязен. Тогда найдутся вершины
	$a, b \in V(G) \setminus S$ в разных компонентах связности;
	следовательно, $S$ является $ab$-разделяющим множеством, и потому
	$\kappa_G(a,b) \le |S| < k$ --- противоречие.
\end{proof}

\begin{Definition}{}{}
	Вершина $v \in V(G)$ называется \emph{точкой сочленения},
	если граф $G \setminus \{v\}$ несвязен.
\end{Definition}

\begin{Definition}{}{}
	Индуцированный подграф $B$ называется \emph{блоком}, если:
	\begin{itemize}
		\item $B$ --- ребро или двусвязный граф;
		\item $B$ максимален: не существует $B' \supsetneq B$,
		удовлетворяющего первому условию.
	\end{itemize}
\end{Definition}

\begin{Statement}{}{}
	Если два различных блока $B_1$ и $B_2$ пересекаются,
	то $|B_1 \cap B_2| = 1$, и единственная общая вершина
	является точкой сочленения.
\end{Statement}
\begin{proof}
	Из максимальности блоков следует, что $B_1 \cup B_2$ не является
	блоком, поэтому $B_1 \cap B_2$ не может содержать более одной
	вершины.
	Пусть $B_1 \cap B_2 = \{v\}$. Если $v$ --- не точка сочленения,
	то между любыми двумя вершинами $B_1 \cup B_2$ есть путь,
	избегающий $v$, то есть $B_1 \cup B_2$ двусвязен --- противоречие
	с максимальностью $B_1$ и $B_2$.
\end{proof}

Обозначим: $A$ --- множество точек сочленения, $\mathcal{B}$ ---
множество блоков графа $G$. Определим \emph{дерево блоков и точек
	сочленения}: его вершины --- элементы $A \cup \mathcal{B}$, а рёбра
соединяют $a \in A$ и $B \in \mathcal{B}$ $\Leftrightarrow$ $a \in V(B)$.

\begin{Theorem}{}{}
	Описанный граф является деревом.
\end{Theorem}
\begin{proof}
	\begin{itemize}
		\item У двух различных блоков не может быть более одной общей вершины:
		иначе их объединение являлось бы блоком, что противоречит максимальности.
		\item Аналогично в графе не может быть цикла из блоков:
		иначе их объединение являлось бы блоком.
	\end{itemize}
\end{proof}
