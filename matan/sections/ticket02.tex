\section{Неравенства. Кривые в пространстве.}

\subsection{Неравенства.}

\begin{Theorem}{Неравенство Гёльдера}{}
    Для $p, q > 0, \ \ \frac{1}{p} + \frac{1}{q} = 1$
    \[
    \int^b_a | (f \cdot g) (x) | dx \le \left(\int^b_a |f(x)|^p dx \right)^\frac{1}{p} \cdot \left(\int^b_a |g(x)|^q dx \right)^\frac{1}{q}
    \]
\end{Theorem}
\begin{proof}

    Из неравенства Юнга:
    \[ 
    a^{\frac{1}{p}} \cdot b^{\frac{1}{q}} \le \frac{a}{p} + \frac{b}{q} \Leftrightarrow 
    a \cdot b \le \frac{a^p}{p} +\frac{b^q}{q}, \qquad a, b > 0 
    \]
    \begin{remark}
        $a$ и $b$ могут принимать $+\infty$, если принято соглашение $+\infty \cdot 0 = 0$
    \end{remark}

    Положим теперь 
    
    $a := \frac{|f(x)|}{\left(\int^b_a |f(x)|^p dx \right)^{\frac{1}{p}}} \qquad$
    $b := \frac{|g(x)|}{\left(\int^b_a |g(x)|^q dx \right)^{\frac{1}{q}}}$

    Случай $\left(\int^b_a |f(x)|^p dx \right)^{\frac{1}{p}} = 0$ рассмотрим позже. Пока что считаем $\not = 0$

    Фиксируем $x \in <a, b>$. Для этого $x$ и наших $a, b$ запишем неравенство Юнга:

    \[
    a \cdot b = \frac{|f(x)| \ \ \cdot \ \ |g(x)|}
    { \left(\int^b_a |f(x)|^p dx \right)^{\frac{1}{p}} \cdot \left( \int^b_a |g(x)|^q dx \right)^{\frac{1}{q}} } \le 
    \]

    Возведем $a^p$ и $b^q$

    \[
    \le \frac{|f(x)|^p}{p\cdot \int^b_a |f(x)|^p dx } +
        \frac{|g(x)|^q}{q\cdot \int^b_a |g(x)|^q dx }
    \]

    Обе части неравенства интегрируемы.

    Проинтегрируем по промежутку $[c, d] \subset \ \ <a, b>$

    \[
    \frac{ \int^d_c |f(x) \ \ \cdot \ \ g(x)| dx } { \left(\int^b_a |f(x)|^p dx \right)^{\frac{1}{p}} \cdot \left( \int^b_a |g(x)|^q dx \right)^{\frac{1}{q}} } \le 
    \frac{\int^d_c |f(x)|^p dx }{p\cdot \int^b_a |f(x)|^p dx } + \frac{\int^d_c |g(x)|^q dx}{q\cdot\int^b_a |g(x)|^q dx }
    \]
    (Интегралы в знаменателе - числа)

    $d \longrightarrow b, c \longrightarrow a:$
    \[
    \frac{ \int^b_a |f(x) \ \ \cdot \ \ g(x)| dx } { \left(\int^b_a |f(x)|^p dx \right)^{\frac{1}{p}} \cdot \left( \int^b_a |g(x)|^q dx \right)^{\frac{1}{q}}}
    \le \frac{1}{p} + \frac{1}{q} = 1
    \]
    $\Rightarrow$
    \[
    \int^b_a |f(x) \cdot g(x)| dx \le  \left(\int^b_a |f(x)|^p dx \right)^{\frac{1}{p}} \cdot \left( \int^b_a |g(x)|^q dx \right)^{\frac{1}{q}}
    \]

    Теперь если один из знаменателей правой части обращается в 0:

    Пусть $\left(\int^b_a |f(x)|^p dx \right)^{\frac{1}{p}} = 0$

    Тогда
    \[
    \int^b_a |f(x) \cdot g(x)| dx = 0
    \]
    
    Доказательство:

    Утверждение:
    \[
    \int^b_a |f(x)|^p dx = 0 \Longrightarrow \int^b_a |f(x)| dx = 0 
    \]

    \[
    \forall [c, d] \subset <a, b>, g - \ \ \text{интегрируема на} \ \ [c, d] \Rightarrow 
    \exists \ \ M(c, d) \st \underset{x \in [c, d]}{sup |g(x)|} \le M(c, d) < \infty
    \]
    \[
    \int^d_c |f(x)g(x)| dx \le M(c, d) \cdot \int^d_c |f(x)| dx \le 
    \] 
    $\le \, M(c, d) \cdot \int^b_a |f(x)| dx$ (увеличили границы интегрирования)

    \[ 0 \le \int^d_c |f(x)g(x)| dx \le M(c, d)\cdot \int^b_a |f(x)| dx = M(c, d) \cdot 0 = 0 \]
    
    \[
    \lim_{c \rightarrow a, d \rightarrow b} \int^d_c |f(x) \cdot g(x)| dx \longrightarrow \int^b_a |f(x) \cdot g(x)| dx = 0
    \]
    ч.т.д.
    \vspace{\baselineskip}

    Неравенство Гельдера выполняется и при интеграле $\rightarrow +\infty$:

    \[
    \left(\int^b_a |f(x)|^p dx \right)^{\frac{1}{p}} = +\infty
    \]
    Тогда правая часть $= +\infty$, которая $\ge$ чего угодно (по соглашению)

\end{proof}

\textbf{Следствие:}

\[
\left(\int^b_a |f(x)| dx\right)^2 \le (b-a) \cdot \int^b_a |f(x)|^2 dx
\]

\begin{proof}
    $g(x) := 1 \ \ \forall x \in <a, b>, \qquad p = q := 2$
    \[ 
    \int_a^b{|f(x)\cdot g(x)| dx} \le \left(\int^b_a |f(x)|^2 dx \right)^\frac{1}{2} \cdot \left(\int^b_a |g(x)|^2 dx \right)^\frac{1}{2} \qquad | \mbox{\textasciicircum} 2
    \]
    \[
    \left(\int^b_a{|f(x)\cdot 1| dx}\right)^2 \le \int^b_a|f(x)|^2 dx \cdot \int^b_a 1 \, dx
    \]
    \[
    \left(\int_a^b{|f(x)| dx}\right)^2 \le \int^b_a|f(x)|^2 dx \cdot (b-a)
    \]
\end{proof}

(Отрезок от a до b конечный. Иначе тоже будет выполнено, но в этом нет смысла так как $+\infty$ больше чего угодно)

\begin{Theorem}{Неравенство Минковского}{}

    Пусть $<a, b> \subset \overline{\RR} = [-\infty; +\infty]$~--- промежуток

    $f, g$~--- интегрируемы на $[c, d] \subset <a, b> \forall c, d$

    Положим: норма $||f||_{L^p} := \left( \int^b_a |f(x)|^p dx \right)^{\frac{1}{p}}$

    $||g||_{L^q} := \left( \int^b_a |g(x)|^q dx \right)^{\frac{1}{q}}$

    $||f||_{L^p} , ||g||_{L^q} \in [0; +\infty]$

    Тогда \[ ||f||_{L^p} + ||g||_{L^p} \ge ||f + g||_{L^p} \]
    для $p \in [1; +\infty)$

\end{Theorem}
Неравенство Минковского по своей сути~--- обобщенное неравенство треугольника.
\begin{proof}

    $p = 1$:

    Получаем
    \[
     \int^b_a |f(x)| dx + \int^b_a | g(x) | dx = \int_{a}^{b} |f(x)| + |g(x)| dx \ge \int^b_a |f + g| (x) dx 
    \]
    

    Теперь пусть $p \in (1; +\infty)$:

    Пусть $||f||_{L^p}, ||g||_{L^p} < +\infty$ (иначе выполнено автоматически)

    Нужно показать, что $||f+g||_{L^p} < +\infty$

    \textbf{Докажем} $(a + b)^p \le 2^{p-1}(a^p+b^p) \qquad a, b \ge 0 - \text{"неравенство на положительные числа"}$:

    \[
    f\left(\frac{a+b}{2}\right) \le \frac{f(a) + f(b)}{2}
    \]
    --- выпуклость.

    $f(t) := t^p, \ \ p > 1 \Rightarrow$ выпуклая вниз на $[0; +\infty)$

    Для двух точек $a, b$ среднее значение $\ge$ реальное значение на выпуклости:

    \[ \frac{a^p}{2} + \frac{b^p}{2} \ge \left( \frac{a + b}{2} \right)^p \]
    \textbf{ч.т.д.}

    Применяя неравенство получаем:

    \[ \forall x \qquad |f(x) + g(x) | ^p \le 2^{p-1}\cdot(|f(x)|^p+|g(x)|^p) \]

    Интегрируя это неравенство на $<a, b>$:

    \[ \int^b_a |f+g|^p(x) dx \le 2^{p-1} \cdot \left(\int_a^b|f(x)|^p dx + \int_a^b |g(x)|^p dx\right) \]

    \[ ||f+g||^p_{L^p} \ \ \le \ \ const \cdot (||f(x)||^p + ||g(x)||^p) < +\infty \]
    ($||f(x)||^p_{L^p} < +\infty, ||g(x)||^p_{L^p} < +\infty$ по условию выше)

    Итак, $||f+g||^p_{L^p} < +\infty$.

    Теперь к доказательству:

    $||f + g||_{L^p} \not = 0$, иначе нечего доказывать

    \[
    ||f+g||^p_{L^p} = \int^b_a |f+g|^p(x) dx = \int^b_a |f+g|^{p-1}(x) \cdot |f+g|(x) dx \overset{\text{нер-во тр-ка}}{\le}  
    \]
    \[
    \le \int^b_a |f+g|^{p-1}(x)\cdot (|f(x)|+|g(x)|) dx = 
    \]
    \[
    = \int^b_a|f(x)|\cdot |f+g|^{p-1}(x) dx + \int^b_a |g(x)||f + g|^{-1}(x) dx
    \]

    Оценим эти интегралы по отдельности:

    Первый интеграл оценивается по неравенству Гельдера: возьмем показатели $p$ и $\frac{1}{1 - \frac{1}{p}}$

    \[ 
    \int^b_a|f(x)|\cdot |f+g|^{p-1}(x) dx \le 
    \]
    \[
    \le \left(\int^b_a \left(|f+g|^{p-1}(x)\right)^{\frac{1}{1 - \frac{1}{p}}} dx \right)^{1-\frac{1}{p}}
    \cdot \left( \int^b_a |f(x)|^p dx \right)^{\frac{1}{p}} =
    \]
    \[
    = \left(\int^b_a |f+g|^p(x) dx \right)^{1-\frac{1}{p}}
    \cdot \left( \int^b_a |f(x)|^p dx \right)^{\frac{1}{p}} = ||f+g||^{p-1}_{L^p}\cdot ||f||_{L^p}
    \]

    Второй интеграл оценивается аналогично:

    \[ \int_a^b|g(x)|\cdot|f+g|^{-1}(x) dx \le ||f+g||_{L^p}^{p-1}\cdot||g||_{L^p}\]
    
    Применяя обе оценки:

    \[
    ||f+g||_{L^p}^p \le ||f+g||^{p-1}_{L^p}\cdot||f||_{L^p} + ||f+g||^{p-1}_{L^p}\cdot||g||_{L^p}
    \]
    \[
    ||f+g||_{L^p}^p \le ||f+g||_{L^p}^{p-1} \cdot (||f||_{L^p} + ||g||_{L^p})
    \]
    \[
     0 < ||f+g||_{L^p} \le ||f||_{L^p}+||g||_{L^p}
    \]

\end{proof}

\begin{Proposition}{Обратное неравенство Минковского}{}

    $<a, b> \subset \overline{\RR} = [-\infty; +\infty]$~--- промежуток.

    $f, g$~--- интегрируемы на $[c, d] \subset <a, b> \forall c, d$

    $\forall \ \ x \ \in <a, b> \st \qquad f(x) \ge 0, g(x) \ge 0$

    $0 < p \le 1$:

    \[ ||f+g||_{L^p} \ge ||f||_{L^p} + ||g||_{L^p}. \]
\end{Proposition}

\begin{proof}

    Аналогично доказательству для прямого неравенства.

    Отметим только, что добавляется условие $f \ge 0, g \ge 0$:

    Это можно заметить из того, что выше мы использовали неравенство треугольника ($|f+g| \le |f|+|g|$).
    Если мы хотим доказать обратное неравенство, то нужен знак в другую сторону($|f+g| \ge |f|+|g|$).
    Ясно, что такое бывает только при одинаковом знаке функций.

    Можно также привести контрпример: $f \not = 0, g = -f$. Получаем $0 >= 2\cdot ||f||_{L^p} > 0$~--- противоречие.
\end{proof}

\begin{remark}
    По индукции несложно показать, что неравенство минковского работает для любого количества слагаемых.
\end{remark}

\begin{Proposition}{Неравенства Гельдера и Минковского для сумм}{}
    
    Пусть $a_k, b_k \ge 0 \qquad k = 1 \ldots N$

    Тогда:

    \begin{enumerate}
        \item \[
            \sum^N_{k=1}a_k \cdot b_k \le \left(\sum^N_{k=1}a^p_k\right)^{\frac{1}{p}}\cdot \left(\sum^N_{k=1}b^q_k \right)^{\frac{1}{q}}, \qquad p, q > 0 \st \frac{1}{p} + \frac{1}{q} = 1
            \]
        \item \[
            \left(\sum^N_{k=1}(a_k + b_k)^p \right)^\frac{1}{p} \le \left(\sum^N_{k=1}(a_k)^p\right)^{\frac{1}{p}} + \left(\sum^N_{k=1}(b_k)^p\right)^{\frac{1}{p}}, \qquad p \in [1, +\infty)
            \]
    \end{enumerate}
    
\end{Proposition}

\begin{proof}

    Пусть $f, g$~--- \textbf{ступенчатые} функции. 

    На каждом отрезке $[k; k+1] f, g$ принимают значения $a_{k+1}, b_{k+1}$ соответсвенно. ($k \in \NN$)

    Тогда на промежутке от $[0; N]$:
    
    \begin{enumerate}
    \item \[
    \left(\int_0^N |f(x)|^p dx \right)^\frac{1}{p} = \left(\sum^N_{k=1}(a_k)^p\right)^{\frac{1}{p}}
    \]
    \item \[
    \left(\int_0^N |g(x)|^q dx \right)^\frac{1}{q} = \left(\sum^N_{k=1}(b_k)^q\right)^{\frac{1}{q}}
    \]
    \item \[
    \int_0^N |f(x) \cdot g(x)| dx = \sum^N_{k=1}(a_k \cdot b_k)
    \]
    \item \[
    \left(\int_0^N |g(x)|^p dx \right)^\frac{1}{p} = \left(\sum^N_{k=1}(b_k)^p\right)^{\frac{1}{p}}
    \]
    \item \[
    \left(\int_0^N |f(x) + g(x)|^p dx \right)^\frac{1}{p} = \left(\sum^N_{k=1}(a_k + b_k)^p \right)^\frac{1}{p}
    \]
    \end{enumerate}
    

    Подставив $(1), (2), (3)$ в неравенство Гёльдера, получим:
    \[
    \sum^N_{k=1}a_k \cdot b_k \le \left(\sum^N_{k=1}a^p_k\right)^{\frac{1}{p}}\cdot \left(\sum^N_{k=1}b^q_k \right)^{\frac{1}{q}}
    \]

    Подставив $(1), (4), (5)$ в неравенство Минковского, получим:
    \[
    \left(\sum^N_{k=1}(a_k + b_k)^p \right)^\frac{1}{p} \le \left(\sum^N_{k=1}(a_k)^p\right)^{\frac{1}{p}} + \left(\sum^N_{k=1}(b_k)^p\right)^{\frac{1}{p}}
    \]

    \begin{remark}
        Сумма~--- интеграл подходяще выверенной функции.
    \end{remark}
\end{proof}

\begin{Theorem}{Неравенство Йенсена}{}

    Пусть $f, g$~--- интегрируемы на $[c, d] \subset <a, b> \forall c, d$ их несобственные интегралы сходятся

    Также пусть $g(x) \ge 0, \int^b_a g(x) dx = 1$ (можно нормировать функцию для такого значения)

    (можно заметить отношение к теории вероятности)

    Пусть $\vphi \st \RR \rightarrow \RR$~--- выпуклая вниз функция. Тогда
    \[
    \vphi\left(\int^b_a g(x)f(x) dx\right) \le \int^b_a \vphi(f(x))g(x) dx
    \]

    ($\vphi$ от мат ожидания $\le$ мат ожидание от $\vphi$)

    В частности, если $<a, b> \subset (-\infty, +\infty):$
    \[
    \vphi\left( \frac{1}{b-a}\cdot\int^b_af(x)dx\right) \le \frac{1}{b-a}\cdot \int^b_a \vphi(f(x))dx
    \]

\end{Theorem}

\begin{proof}

    На самом деле доказываем:
    $a_k \ge 0, \qquad x=1\ldots N, \qquad \sum^N_{k=1}a_k > 0$

    $-\infty < x_1 \le x_2 \le \ldots \le x_n < +\infty$

    Сразу для сумм 
    \[ 
    \vphi\left(\frac{\sum^N_{k=1}a_kx_k}{\sum^N_{k=1}a_k}\right) \le \frac{\sum^N_{k=1}a_k\vphi(x_k)}{\sum^N_{k=1}a_k}
    \]

    \[
    \alpha_k := \frac{a_k}{\sum^N_{k=1}a_k}
    \] 
    тогда $\alpha_k \ge 0$~--- выпуклая комбинация. $\sum^N_{k=1}\alpha_k = 1$

    $\vphi \left(\sum^N_{k=1}\alpha_kx_k\right) \le \sum^N_{k=1}\alpha_k\vphi(x_k)$

    Вследствие выпуклости: среднее значение $ = \sum\alpha_k\vphi(x_k)$,
    реальное значение $= \vphi(\sum\alpha_kx_k)$

    (можно доказать графиком, начертив выпуклую функцию, либо по индукции)

    Перейдем к пределу и получим неравенство

    \[
    \vphi\left(\sum^N_{k=1}\alpha_kx_k\right) \le \sum^N_{k=1}\alpha_k \vphi(x_k) \Rightarrow 
    \vphi\left(\int^b_a g(x)f(x) dx\right) \le \int^b_a \vphi(f(x))g(x) dx 
    \]
\end{proof}



\subsection{Кривые в пространстве.}

Рассмотрим вопросы:

\begin{itemize}
    \item Что такое длина кривой $\gamma$?
    \item Что такое кривая $\gamma$?
\end{itemize}

$\RR^d = \{ \left( x_1, \ldots, x_d \right) \st x_k \in \RR, \ \ k = 1, \ldots, d \} \qquad d \in \NN$

Введем метрику:

$x, y \in \RR^d. x = \left(x_1, \ldots, x_d \right),  y = \left(y_1, \ldots, y_d \right)$
\[
|x - y| := \sqrt{\sum^d_{k=1}(x_k - y_k)^2}
\]

\begin{proof}
    \begin{itemize}
        \item $\ge 0$ так как $\sqrt{()} \ge 0$
        \item $|y - x| = \sqrt{\sum^d_{k=1}(y_k - x_k)^2} = \sqrt{\sum^d_{k=1}(x_k - y_k)^2} = |x - y|$
        \item $|x - y| = 0 \Leftrightarrow \sum^d_{k=1}(x_k - y_k)^2 = 0 \Leftrightarrow (x_k - y_k)^2 = 0 \ \ \forall \ \ k$
        
        $\Leftrightarrow x_k = y_k \ \ \forall k \Leftrightarrow x = y$
        \item $|x - y| + |y - z| \ge |x - z|$
        \[
        a_k := x_k - y_k, \; b_k := y_k - z_k \Rightarrow a_k + b_k = x_k - z_k.
        \]

        Тогда для последовательностей $a_k, b_k$ по неравенству Минковского для сумм при $p = 2$ получим:

        \[
        \left(\sum^N_{k=1}(x_k-z_k)^2 \right)^\frac{1}{2} \le \left(\sum^N_{k=1}(x_k-y_k)^2\right)^{\frac{1}{2}} + \left(\sum^N_{k=1}(y_k-z_k)^2\right)^{\frac{1}{2}}
        \]

        Что по определению
        \[
        |x - z| \le |x - y| + |y - z|
        \]
        
    \end{itemize}
\end{proof}

\begin{Definition}{Кривая в $\RR^d$}{}

    Кривая в $\RR^d$~--- множество точек $\Gamma$ в пространстве $\RR^d$,
    
    $\Gamma = \{ \vphi_1(t), \ldots, \vphi_d(t) \}_{t \in I} \qquad I \subset \RR$~--- промежуток.

    $\vphi_k \st I \longrightarrow \RR$~--- непрерывна.
    
    $\vphi_k$~--- \textbf{параметризация} точек.
\end{Definition}

\begin{Definition}{Кратная точка}{}

    Точка $x \in \RR^d$~--- кратная точка кривой, если $\exists \ \ \ge 2$ точек $t \in I \st$
     $t_1 \not = t_2$
     \[ x = \vphi(t_1)=\vphi(t_2); \qquad \vphi = (\vphi_1, \ldots, \vphi_d) \]
\end{Definition}

\begin{Definition}{}{}

    Кривая без кратных точек (не учитывая концы)~--- \textbf{простая} кривая.

    Если концы совпадают~--- \textbf{замкнутая}.

    Если конечное число кратных точек, то \textbf{параметризуемая}.
\end{Definition}

\begin{Definition}{Ломаная}{}

    \textbf{Ломаная}~--- кривая, что отрезок $I = [t_0; t_N]$ \textbf{разбивается} на конечное число отрезков, таких что

    \[
    \forall \ \ k \st \vphi_k(t) \qquad t_j \le t \le t_{j+1}
    \]
    --- \textbf{линейная} функция.
\end{Definition}

\begin{remark}
    Звенья ломаной понимаем интуитивно.
\end{remark}

\begin{Definition}{Вписанная ломаная}{}

    $\Gamma$~--- кривая в $\RR^d$

    $\Gamma'$~--- ломаная, соотвествующая разбиению $\tau = \{ t_0, t_1, \ldots, t_N \}$ отрезка $I$.

    $\Gamma'$~--- \textbf{вписана} в $\Gamma$, если 
    \[ \widetilde{\vphi}{(t_j)} = \vphi (t_j ) \qquad 0 \le j \le N \] 

    \vspace{1.5\baselineskip}
    Длина ломаной $\Gamma'$ = длина звеньев $\Gamma' = |\Gamma'| = l(\Gamma') = \Lambda_l(\Gamma')$

    \[
    l(\Gamma') := \sum^{N-1}_{j=0}{\sqrt{\sum^d_{k=1}{\left[\widetilde{\vphi_k}(t_j) - \widetilde{\vphi_k}(t_{j+1})\right]^2}}}
    \]

\end{Definition}

\begin{Definition}{Спрямляемая прямая}{}
    Кривая $\Gamma$~--- \textbf{спрямляемая}, если
    \[
    \sup \{ l(\Gamma') \st \Gamma' - \text{вписаная в } \ \ \Gamma \} < +\infty
    \]
\end{Definition}

Пример:

$I = (0, 1], \qquad \vphi(t) = (t, \sin{\frac{1}{t}})$~--- не спрямляема.


\begin{Definition}{}{}
    Если $\Gamma$~--- спрямляемая кривая в $\RR^d$, то 
    \[
    l(\Gamma) := \sup \{ l(\Gamma') \st \Gamma' - \text{ломаная, вписанная в } \ \ \Gamma \}
    \]
\end{Definition}


\begin{Theorem}{Длина спрямляемой кривой}{}
    Пусть $\Gamma$~--- гладкая кривая (можем брать производные) в $\RR^d$, то есть 

    $\Gamma=\vphi(I),\ \ I=[a,b]\ \ \vphi=(\vphi_1, \ldots, \vphi_d)\ \ $
    $\vphi_n: I \rightarrow \RR,\ \ 1 \le n \le d$

    и $\vphi'_n \in C([a, b])$. На концах подразумевается односторонняя производная:

    \[ \vphi'_n(a) = \lim_{t \rightarrow a+} \frac{\vphi_n(a) - \vphi_n(t)}{a-t} \]
    \[ \vphi'_n(b) = \lim_{t \rightarrow a-} \frac{\vphi_n(a) - \vphi_n(t)}{a-t} \]

    Тогда $\Gamma$~--- спрямляется и 
    \[ 
    l(\Gamma)=\int^b_a \sqrt{\sum^d_{n=1} (\vphi'_n(t))^2} dt 
    \]
\end{Theorem}

\begin{remark}
    Важно, чтобы ломаная была \textbf{вписанная} в кривую, иначе легко можно получить неверный результат или бесконечность.

    Вообще идеологически правильнее, выбрав точки на кривой, построить касательные в этих точках и суммировать отрезки касательных,
    устремив мелкость $\rightarrow 0$. Но сильно проще расписать выражение для ломаной, поэтому рассмотрим именно такой подход.
\end{remark}

Далее все $\Gamma'$ считаем вписанными в кривую $\Gamma$ ломаными.

\begin{proof}

    Положим
    \[
    L := \int_{a}^{b}{\sqrt{\sum_{n=1}^d{(\vphi_n'(t))^2}} \ \ dt}
    \]

    Докажем два неравенства:

    \begin{enumerate}
        \item Покажем, что $\ \ \forall \ \ \Gamma' \st l(\Gamma') \le L$.
    
        $\Rightarrow \sup \{l(\Gamma')\} = l(\Gamma) \le L$.
        \item Явно предъявим последовательность $\{l(\Gamma'_k)\}$ такую, что $\lim_{k \rightarrow +\infty}l(\Gamma') = L$:
        
        Для $\{l(\Gamma'_k)\} \subset \{l(\Gamma')\} \qquad \sup \{l(\Gamma'_k)\} = L$

        $\Rightarrow \sup \{l(\Gamma')\} = l(\Gamma) \ge L$. 
    \end{enumerate}
    Из этого получим $l(\Gamma) = L$.

    \vspace{2\baselineskip}
    
    $\ \ \forall \ \ \Gamma' \st l(\Gamma') \le L$:

    Пусть $\{t_j\}_1^N$~--- точки параметризации ломаной.

    \[ l(\Gamma') \overset{(!)}{\le} L \Leftrightarrow \]
    \[ 
    \sum_{j=1}^{N-1} \sqrt{\sum_{n=1}^d (\vphi_n(t_{j+1}) - \vphi_n(t_j))^2} \qquad \overset{(!)}{\le} \qquad \int_{a}^{b}{\sqrt{\sum_{n=1}^d{(\vphi_n'(t))^2}} \ \ dt} 
    = \sum_{j=1}^{N-1} \int_{t_j}^{t_j+1}{\sqrt{\sum_{n=1}^d{(\vphi_n'(t))^2}} \ \ dt} 
    \]
    \[ 
    \Leftarrow \sqrt{\sum_{n=1}^d (\vphi_n(t_{j+1}) - \vphi_n(t_j))^2} \qquad \overset{(!)}{\le} \qquad \int_{t_j}^{t_{j+1}}{\sqrt{\sum_{n=1}^d{(\vphi_n'(t))^2}} \ \ dt} 
    \]
    \[ 
    \Leftrightarrow \sum_{n=1}^d (\vphi_n(t_{j+1}) - \vphi_n(t_j))^2 \qquad \overset{(!)}{\le} \qquad \left(\int_{t_j}^{t_{j+1}}{\sqrt{\sum_{n=1}^d{(\vphi_n'(t))^2}} \ \ dt} \right)^2
    \]
    По формуле Ньютона-Лейбница $\vphi_n(t_{j+1}) - \vphi_n(t_j) = \int_{t_j}^{t_{j+1}} \vphi_n'(t) \ \ dt$

    \[ 
    \Leftrightarrow \sum_{n=1}^d \left(\int_{t_j}^{t_{j+1}} \vphi_n'(t) \ \ dt\right)^2 \qquad \overset{(!)}{\le} \qquad \left(\int_{t_j}^{t_{j+1}}{\sqrt{\sum_{n=1}^d{(\vphi_n'(t))^2}} \ \ dt} \right)^2
    \]

    Пусть $f_i(t) := (\vphi_i'(t))^2$

    \[ 
    \Leftarrow \sum_{n=1}^d \left(\int_{t_j}^{t_{j+1}} (f_n(t))^\frac{1}{2} \ \ dt\right)^2 \qquad \overset{(!)}{\le} \qquad \left(\int_{t_j}^{t_{j+1}}{ \left( \sum_{n=1}^d f_n(t) \right)^\frac{1}{2}} \ \ dt  \right)^2
    \]

    $f_i = (\vphi_i')^2 \ge 0$

    Последнее неравенство выполнено, так как удовлетворяет условию обратного неравенства Минковского для $p = \frac{1}{2}$.

    Доказанное неравенство сильнее требуемого изначально, так как:

    \[
    \sum_{n=1}^d \left(\int_{t_j}^{t_{j+1}} \vphi_n'(t) \ \ dt\right)^2 \le \sum_{n=1}^d \left(\int_{t_j}^{t_{j+1}} |\vphi_n'(t)| \ \ dt\right)^2 = \sum_{n=1}^d \left(\int_{t_j}^{t_{j+1}} (f_n(t))^\frac{1}{2} \ \ dt\right)^2
    \]

    \vspace{1\baselineskip}
    Итак, $l(\Gamma') \le L \ \ \Leftrightarrow \ \ \sup \{l(\Gamma')\} = l(\Gamma) \le L$
    \vspace{2\baselineskip}

    Теперь построим последовательность $\{l(\Gamma'_k)\}$, предел который будет равен $L$.

    Возьмем диадическое дробление отрезка параметризации $I$ ранга $k$:

    $j = 0, \ldots, N_k$

    $\{t_{k, j}\}$~--- дробление отрезка параметризации $I = [a, b]$

    \[
    \Gamma'_k = \{\vphi_1(t_{k, j}), \ldots, \vphi_d(t_{k, j}) \}_{j=0}^{N_k} \in \Gamma = \vphi([a, b])
    \]

    Ломаную $\Gamma'_k$ построим так: 

    Берем две последовательные точки $\vphi(t_{k, j}), \vphi(t_{k, j+1})$, соединяем их отрезком прямой.

    По построению: $l(\Gamma'_k) \le l(\Gamma) \; \; \forall \ \ k \in \NN \qquad$ (потому что каждое звено ломаной меньше длины стягиваемой хорды)
    

    Проверим, что 
    \[
    l(\Gamma'_k) \overset{k \rightarrow +\infty}{\longrightarrow} \int_{a}^{b} \sqrt{\sum_{n=1}^d{|\vphi'_n|^2(x)}} dx
    \]

    Оценим $l(\Gamma'_k)$:

    $\vphi_k'$ непрерывна на отрезке $[a, b]$ (в концах разумеется производная односторонняя)

    $\Rightarrow$ функции равномерно непрерывны $\Rightarrow$
    
    \[ \forall \ \ \eps > 0 \ \ \exists \ \ k_0 \st \forall \ \ k \ge k_0 \ \
     \forall \ \ [t_{k, j}; t_{k, j+1}] \ \ \forall \ \ \tau_1, \tau_2 \in [t_{k, j}; t_{k, j+1}] \]

    выполняется:

    \[
    |\vphi_n'(\tau_1) - \vphi_n'(\tau_2)| < \eps, \qquad \forall \ \ n = 1 \ldots d
    \]

    Сумма длин звеньев ломаной
    \[ l(\Gamma'_k) = \sum_{j=0}^{N_k-1}{\sqrt{\sum_{n=1}^d{(\vphi_n(t_{k, j+1}) - \vphi_n(t_{k, j}))^2}} \ \ dt } \]

    Оценим разность $L - l(\Gamma'_k)$:

    \[
    \int_a^b{\sqrt{\sum_{n=1}^d{|\vphi_n'|^2(t)}} \ \ dt} - l(\Gamma_k') =
    \]
    \vspace{\baselineskip}
    \[
    = \int_a^b{\sqrt{\sum_{n=1}^d{|\vphi_n'|^2(t)}} \ \ dt} - 
    \sum_{j=0}^{N_k-1}\sqrt{\sum_{n=1}^d{(\vphi_n(t_{k, j+1}) - \vphi_n(t_{k, j}))^2}} = 
    \]
    \vspace{\baselineskip}
    \[
    = \sum_{j=0}^{N_k-1} \int_{t_{k, j}}^{t_{k,j+1}}{\sqrt{\sum_{n=1}^d{|\vphi_n'|^2(t)}}\ \ dt} -
    \sum_{j=0}^{N_k-1}\left(|t_{k, j+1} - t_{k, j}| \cdot \sqrt{\frac{\sum_{n=1}^d{(\vphi_n(t_{k, j+1}) - \vphi_n(t_{k, j}))^2}}{|t_{k, j+1} - t_{k, j}|^2}}\right) =
    \]

    \vspace{\baselineskip}
    $|t_{k, j+1} - t_{k, j}| = \int_{t_{k, j}}^{t_{k,j+1}} 1 \ \ dt$ 
    \vspace{\baselineskip}
    \[
    = \sum_{j=0}^{N_k-1}\int_{t_{k, j}}^{t_{k,j+1}}\left( \sqrt{\sum_{n=1}^d{|\vphi_n'|^2(t)}} - \sqrt{\sum_{n=1}^d \frac{{(\vphi_n(t_{k, j+1}) - \vphi_n(t_{k, j}))^2}}{|t_{k, j+1} - t_{k, j}|^2}} \right) dt \le
    \]

    $p := 2$, применим неравенство Минковского для сумм:
    \vspace{\baselineskip}

    $|\sqrt{\sum_{n=1}^d\alpha_n^2} - \sqrt{\sum_{n=1}^d\beta_n^2}| \le \sqrt{\sum_{n=1}^d(\alpha_n - \beta_n)^2}$ 
    \vspace{\baselineskip}
    \[
    \le \sum_{j=0}^{N_k-1}\int_{t_{k, j}}^{t_{k,j+1}} \sqrt{\sum_{k=1}^d \left( |\vphi_n'|^2(t) -  \frac{(\vphi_n(t_{k, j+1}) - \vphi_n(t_{k, j}))^2}{|t_{k, j+1} - t_{k, j}|^2} \right) } =
    \]

    По теореме Коши для производных на отрезке $[t_{k, j}; t_{k,j+1}]$:

    \[
    \frac{(\vphi_n(t_{k, j+1}) - \vphi_n(t_{k, j}))}{|t_{k, j+1} - t_{k, j}|} = \vphi_n'(\tau_{k,j}^n) \qquad \tau_{k,j}^n \in [t_{k, j}; t_{k,j+1}]
    \]    
    \vspace{\baselineskip}
    \[
    = \sum_{j=0}^{N_k-1}\int_{t_{k, j}}^{t_{k,j+1}} \sqrt{\sum_{n=1}^d \left( \vphi_n'(t) - \vphi'(\tau_{k,j}^n) \right)^2} dt \overset{k \ge k_0(\eps)}{\le}
    \]

    \[
    \le \sum_{j=0}^{N_k-1}\int_{t_{k, j}}^{t_{k,j+1}} \sqrt{\sum_{n=1}^d \left( \eps \right)^2} dt =
    \sum_{j=0}^{N_k-1}\int_{t_{k, j}}^{t_{k,j+1}} \eps \cdot \sqrt{\sum_{n=1}^d 1} \ \ dt =
    \]
    \vspace{\baselineskip}
    \[
    \sum_{j=0}^{N_k-1}\eps \cdot \int_{t_{k, j}}^{t_{k,j+1}}\sqrt{d} = \sum_{j=0}^{N_k-1} \eps \cdot  \sqrt{d} \cdot |t_{k, j+1}-t_{k,j}| = \eps \cdot  \sqrt{d} \cdot (b-a)
    \]

    \[
    = (\sqrt{d} \cdot (b-a)) \cdot \eps = const \cdot \eps \overset{k \rightarrow +\infty}{\longrightarrow} 0
    \]
    А значит, $\lim_{k \rightarrow +\infty} l(\Gamma'_k) = L$

    \vspace{\baselineskip}

    \[
    L \le l(\Gamma) \le L \Rightarrow
    \]
    \[
    l(\Gamma) = L = \int_a^b (\sqrt{\sum_{n=1}^d |\vphi_n'(t)|^2}\ \ dt)
    \]

\end{proof}

\begin{Example}{Параметризация окружности}{}

    Для оркужностей радиуса $r > 0$:

    \[
    \vphi_1(t) = r \cdot \cos{2\pi t} \qquad \vphi_2(t) = r \cdot \sin{2\pi t}, \qquad t \in [0; 1]
    \]

    $\pi$ можно понимать как абстрактное число, полученное, например, из формулы Валлиса.

    Тогда:
    \[
    l(\Gamma) = \int_{0}^{1} \sqrt{r^2(2\pi)^2\cdot((-\sin)^2(2\pi t) + \cos^2(2\pi t))} \ \ dt = 2\pi r.
    \]
\end{Example}
