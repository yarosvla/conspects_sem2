\section{Неравенства. Кривые в пространстве.}

\subsection{Неравенства.}

\begin{Theorem}{Неравенство Гёльдера}{}
	Пусть $f,g:(a,b)\to\mathbb{R}$, и $f,g$ интегрируемы по Риману на $[c,d]\subset(a,b)$ для любого $[c,d]$.
	
	\medskip
	\textbf{Тогда}
	\[
	\int_a^b \bigl|f(x)\,g(x)\bigr|\,dx
	\le
	\left(\int_a^b |f(x)|^{p}\,dx\right)^{\!\frac1p}
	\cdot
	\left(\int_a^b |g(x)|^{q}\,dx\right)^{\!\frac1q},
	\]
	\[
	\forall\, p,q>0,\ p,q<\infty:\qquad \frac1p+\frac1q=1.
	\]
	
	\medskip
	Если $p=q=2$, то имеем неравенство КБШ:
	\[
	\int_a^b |fg|
	\le
	\sqrt{\int_a^b |f|^{2}}\;\cdot\;\sqrt{\int_a^b |g|^{2}}.
	\]
\end{Theorem}

\begin{proof}
	Исходим из неравенства Юнга:
	\[
	a^{\frac{1}{p}} \cdot b^{\frac{1}{q}} \le \frac{a}{p} + \frac{b}{q}
	\qquad \Longleftrightarrow \qquad
	a \cdot b \le \frac{a^p}{p} + \frac{b^q}{q},
	\qquad a,b>0.
	\]
	
	\begin{remark}
		Здесь $a$ и $b$ могут принимать значение $+\infty$, если принято соглашение $+\infty \cdot 0 = 0$.
	\end{remark}
	
	Положим
	\[
	A:=\left(\int_a^b |f(x)|^p\,dx\right)^{\frac{1}{p}},
	\qquad
	B:=\left(\int_a^b |g(x)|^q\,dx\right)^{\frac{1}{q}}.
	\]
	Случаи $A=0$ или $B=0$ рассмотрим позже. Пока считаем, что $A\ne 0$ и $B\ne 0$.
	
	Зафиксируем $x\in(a,b)$ и применим неравенство Юнга к числам
	\[
	a:=\frac{|f(x)|}{A},
	\qquad
	b:=\frac{|g(x)|}{B}.
	\]
	Тогда
	\[
	\frac{|f(x)|\,|g(x)|}{A\cdot B}
	\le
	\frac{|f(x)|^p}{p\int_a^b |f(x)|^p\,dx}
	+
	\frac{|g(x)|^q}{q\int_a^b |g(x)|^q\,dx}.
	\]
	
	Обе части этого неравенства интегрируемы. Проинтегрируем по промежутку $[c,d]\subset(a,b)$:
	\[
	\frac{\int_c^d |f(x)g(x)|\,dx}{A\cdot B}
	\le
	\frac{\int_c^d |f(x)|^p\,dx}{p\int_a^b |f(x)|^p\,dx}
	+
	\frac{\int_c^d |g(x)|^q\,dx}{q\int_a^b |g(x)|^q\,dx}.
	\]
	(Интегралы в знаменателях --- числа.)
	
	Теперь устремим $c\to a$, $d\to b$. Получаем
	\[
	\frac{\int_a^b |f(x)g(x)|\,dx}{A\cdot B}
	\le
	\frac{1}{p}+\frac{1}{q}=1.
	\]
	Следовательно,
	\[
	\int_a^b |f(x)g(x)|\,dx
	\le
	\left(\int_a^b |f(x)|^p\,dx\right)^{\frac{1}{p}}
	\cdot
	\left(\int_a^b |g(x)|^q\,dx\right)^{\frac{1}{q}}.
	\]
	
	Теперь рассмотрим случай, когда один из множителей в правой части обращается в $0$.
	
	Пусть
	\[
	\left(\int_a^b |f(x)|^p\,dx\right)^{\frac{1}{p}}=0.
	\]
	Тогда
	\[
	\int_a^b |f(x)g(x)|\,dx=0.
	\]
	
	\textbf{Доказательство}. Используем утверждение:
	\[
	\int_a^b |f(x)|^p\,dx=0 \Longrightarrow \int_a^b |f(x)|\,dx=0.
	\]
	Кроме того, для любого $[c,d]\subset(a,b)$ функция $g$ интегрируема на $[c,d]$, а значит, ограничена на этом отрезке:
	\[
	\exists\, M(c,d)\ \ \text{такое, что}\ \ \sup_{x\in[c,d]} |g(x)| \le M(c,d) < \infty.
	\]
	Поэтому
	\[
	\int_c^d |f(x)g(x)|\,dx
	\le
	M(c,d)\cdot \int_c^d |f(x)|\,dx
	\le
	M(c,d)\cdot \int_a^b |f(x)|\,dx.
	\]
	Отсюда
	\[
	0 \le \int_c^d |f(x)g(x)|\,dx \le M(c,d)\cdot \int_a^b |f(x)|\,dx = M(c,d)\cdot 0 = 0.
	\]
	Значит,
	\[
	\int_c^d |f(x)g(x)|\,dx=0.
	\]
	Переходя к пределу при $c\to a$, $d\to b$, получаем
	\[
	\int_a^b |f(x)g(x)|\,dx=0.
	\]
	Что и требовалось.
	
	\medskip
	Неравенство Гёльдера выполняется и в случае, когда один из интегралов в правой части равен $+\infty$. Например, если
	\[
	\left(\int_a^b |f(x)|^p\,dx\right)^{\frac{1}{p}}=+\infty,
	\]
	то правая часть равна $+\infty$, а значит, неравенство выполнено по соглашению.
\end{proof}

\begin{Corollary}{}{}
\[
\left(\int_a^b |f(x)|\,dx\right)^2 \le (b-a)\cdot \int_a^b |f(x)|^2\,dx.
\]
\end{Corollary}

\begin{proof}
	Положим $g(x):=1$ для всех $x\in(a,b)$ и возьмем $p=q=2$. Тогда по неравенству Гёльдера
	\[
	\int_a^b |f(x)\cdot g(x)|\,dx
	\le
	\left(\int_a^b |f(x)|^2\,dx\right)^{\frac{1}{2}}
	\cdot
	\left(\int_a^b |g(x)|^2\,dx\right)^{\frac{1}{2}}.
	\]
	Возведем обе части в квадрат:
	\[
	\left(\int_a^b |f(x)\cdot 1|\,dx\right)^2
	\le
	\int_a^b |f(x)|^2\,dx \cdot \int_a^b 1\,dx.
	\]
	Следовательно,
	\[
	\left(\int_a^b |f(x)|\,dx\right)^2
	\le
	\int_a^b |f(x)|^2\,dx \cdot (b-a).
	\]
\end{proof}

(Отрезок от $a$ до $b$ конечный. Иначе неравенство тоже будет выполнено, но в этом нет смысла, так как $+\infty$ больше чего угодно.)

\begin{Theorem}{Неравенство Минковского}{}
	Пусть $<a,b>\subset \overline{\RR}=[-\infty;+\infty]$ --- промежуток.
	
	Пусть $f,g$ интегрируемы на $[c,d]\subset<a,b>$ для любых $c,d$.
	
	Положим
	\[
	||f||_{L^p}:=\left(\int_a^b |f(x)|^p\,dx\right)^{\frac{1}{p}},
	\qquad
	||g||_{L^p}:=\left(\int_a^b |g(x)|^p\,dx\right)^{\frac{1}{p}}.
	\]
	Тогда
	\[
	||f||_{L^p},\,||g||_{L^p}\in[0;+\infty].
	\]
	
	Для $p\in[1;+\infty)$ верно:
	\[
	||f||_{L^p}+||g||_{L^p}\ge ||f+g||_{L^p}.
	\]
\end{Theorem}

Неравенство Минковского по своей сути --- обобщенное неравенство треугольника.

\begin{proof}
	Сначала рассмотрим случай $p=1$. Тогда
	\[
	\int_a^b |f(x)|\,dx + \int_a^b |g(x)|\,dx
	=
	\int_a^b \bigl(|f(x)|+|g(x)|\bigr)\,dx
	\ge
	\int_a^b |f(x)+g(x)|\,dx.
	\]
	
	Теперь пусть $p\in(1;+\infty)$.
	
	Если хотя бы одна из норм $||f||_{L^p}$, $||g||_{L^p}$ равна $+\infty$, то неравенство выполнено автоматически. Поэтому будем считать, что
	\[
	||f||_{L^p}<+\infty,
	\qquad
	||g||_{L^p}<+\infty.
	\]
	
	Сначала покажем, что тогда $||f+g||_{L^p}<+\infty$.
	
	\textbf{Докажем} вспомогательное неравенство
	\[
	(a+b)^p \le 2^{p-1}(a^p+b^p),
	\qquad a,b\ge 0.
	\]
	Действительно, функция $t\mapsto t^p$ при $p>1$ выпукла вниз на $[0;+\infty)$, поэтому
	\[
	\left(\frac{a+b}{2}\right)^p \le \frac{a^p+b^p}{2}.
	\]
	Умножая на $2^p$, получаем требуемое:
	\[
	(a+b)^p \le 2^{p-1}(a^p+b^p).
	\]
	
	Применяя это неравенство, получаем:
	\[
	\forall x \qquad |f(x)+g(x)|^p \le 2^{p-1}\bigl(|f(x)|^p+|g(x)|^p\bigr).
	\]
	Интегрируя по $<a,b>$, имеем
	\[
	\int_a^b |f(x)+g(x)|^p\,dx
	\le
	2^{p-1}\left(\int_a^b |f(x)|^p\,dx + \int_a^b |g(x)|^p\,dx\right).
	\]
	Следовательно,
	\[
	||f+g||_{L^p}^p
	\le
	\mathrm{const}\cdot\bigl(||f||_{L^p}^p+||g||_{L^p}^p\bigr)<+\infty.
	\]
	
	Итак, $||f+g||_{L^p}<+\infty$.
	
	Теперь переходим к доказательству самого неравенства. Если
	\[
	||f+g||_{L^p}=0,
	\]
	то доказывать нечего. Пусть поэтому $||f+g||_{L^p}\ne 0$.
	
	Тогда
	\[
	||f+g||_{L^p}^p
	=
	\int_a^b |f+g|^p(x)\,dx
	=
	\int_a^b |f+g|^{p-1}(x)\cdot |f+g|(x)\,dx.
	\]
	По неравенству треугольника
	\[
	|f+g|(x)\le |f(x)|+|g(x)|,
	\]
	поэтому
	\[
	||f+g||_{L^p}^p
	\le
	\int_a^b |f+g|^{p-1}(x)\bigl(|f(x)|+|g(x)|\bigr)\,dx.
	\]
	Раскрывая скобки, получаем
	\[
	||f+g||_{L^p}^p
	\le
	\int_a^b |f(x)|\cdot |f+g|^{p-1}(x)\,dx
	+
	\int_a^b |g(x)|\cdot |f+g|^{p-1}(x)\,dx.
	\]
	
	Оценим эти интегралы по отдельности.
	
	Для первого интеграла применим неравенство Гёльдера с показателями $p$ и $\frac{1}{1-\frac{1}{p}}$:
	\[
	\int_a^b |f(x)|\cdot |f+g|^{p-1}(x)\,dx
	\le
	\left(\int_a^b \left(|f+g|^{p-1}(x)\right)^{\frac{1}{1-\frac{1}{p}}}\,dx\right)^{1-\frac{1}{p}}
	\cdot
	\left(\int_a^b |f(x)|^p\,dx\right)^{\frac{1}{p}}.
	\]
	Так как
	\[
	(p-1)\cdot \frac{1}{1-\frac{1}{p}} = p,
	\]
	то
	\[
	\int_a^b |f(x)|\cdot |f+g|^{p-1}(x)\,dx
	\le
	\left(\int_a^b |f+g|^p(x)\,dx\right)^{1-\frac{1}{p}}
	\cdot
	\left(\int_a^b |f(x)|^p\,dx\right)^{\frac{1}{p}}.
	\]
	Значит,
	\[
	\int_a^b |f(x)|\cdot |f+g|^{p-1}(x)\,dx
	\le
	||f+g||_{L^p}^{p-1}\cdot ||f||_{L^p}.
	\]
	
	Аналогично
	\[
	\int_a^b |g(x)|\cdot |f+g|^{p-1}(x)\,dx
	\le
	||f+g||_{L^p}^{p-1}\cdot ||g||_{L^p}.
	\]
	
	Подставляя обе оценки, получаем
	\[
	||f+g||_{L^p}^p
	\le
	||f+g||_{L^p}^{p-1}\cdot ||f||_{L^p}
	+
	||f+g||_{L^p}^{p-1}\cdot ||g||_{L^p},
	\]
	то есть
	\[
	||f+g||_{L^p}^p
	\le
	||f+g||_{L^p}^{p-1}\bigl(||f||_{L^p}+||g||_{L^p}\bigr).
	\]
	Делим на $||f+g||_{L^p}^{p-1}>0$ и получаем
	\[
	0<||f+g||_{L^p}\le ||f||_{L^p}+||g||_{L^p}.
	\]
\end{proof}

\begin{Proposition}{Обратное неравенство Минковского}{}
	$<a,b>\subset \overline{\RR}=[-\infty;+\infty]$ --- промежуток.
	
	Пусть $f,g$ интегрируемы на $[c,d]\subset<a,b>$ для любых $c,d$.
	
	Пусть
	\[
	\forall x\in<a,b> \qquad f(x)\ge 0,\quad g(x)\ge 0.
	\]
	Тогда при $0<p\le 1$:
	\[
	||f+g||_{L^p}\ge ||f||_{L^p}+||g||_{L^p}.
	\]
\end{Proposition}

\begin{proof}
	Доказательство аналогично доказательству прямого неравенства.
	
	Отметим только, что здесь существенно используется условие $f\ge 0$, $g\ge 0$.
	
	Действительно, выше мы использовали неравенство треугольника
	\[
	|f+g|\le |f|+|g|.
	\]
	Если же хотим получить обратное неравенство, то нужен знак в другую сторону. Ясно, что это возможно только при одинаковом знаке функций.
	
	Можно также привести контрпример: $f\ne 0$, $g=-f$. Тогда получили бы
	\[
	0 \ge 2\cdot ||f||_{L^p} > 0,
	\]
	что невозможно.
\end{proof}

\begin{remark}
	По индукции несложно показать, что неравенство Минковского работает для любого количества слагаемых.
\end{remark}

Рассмотрим частный случай интегралов --- суммы.

\begin{Proposition}{Неравенства Гёльдера и Минковского для сумм}{}
	Пусть $a_k,b_k\ge 0$, \qquad $k=1,\ldots,N$.
	
	Тогда:
	\begin{enumerate}
		\item
		\[
		\sum_{k=1}^N a_k b_k
		\le
		\left(\sum_{k=1}^N a_k^p\right)^{\frac{1}{p}}
		\cdot
		\left(\sum_{k=1}^N b_k^q\right)^{\frac{1}{q}},
		\qquad
		p,q>0,\ \ \frac{1}{p}+\frac{1}{q}=1.
		\]
		\item
		\[
		\left(\sum_{k=1}^N (a_k+b_k)^p\right)^{\frac{1}{p}}
		\le
		\left(\sum_{k=1}^N a_k^p\right)^{\frac{1}{p}}
		+
		\left(\sum_{k=1}^N b_k^p\right)^{\frac{1}{p}},
		\qquad
		p\in[1,+\infty).
		\]
	\end{enumerate}
\end{Proposition}

\begin{proof}
	Пусть $f,g$ --- \textbf{ступенчатые} функции.
	
	На каждом отрезке $[k,k+1]$ функции $f,g$ принимают значения $a_{k+1}, b_{k+1}$ соответственно ($k\in\NN$).
	
	Тогда на промежутке $[0,N]$:
	\begin{equation}
		\left(\int_0^N |f(x)|^p\,dx\right)^{\frac{1}{p}}
		=
		\left(\sum_{k=1}^N a_k^p\right)^{\frac{1}{p}}.
	\end{equation}
	
	\begin{equation}
		\left(\int_0^N |g(x)|^q\,dx\right)^{\frac{1}{q}}
		=
		\left(\sum_{k=1}^N b_k^q\right)^{\frac{1}{q}}.
	\end{equation}
	
	\begin{equation}
		\int_0^N |f(x)\cdot g(x)|\,dx
		=
		\sum_{k=1}^N a_k b_k.
	\end{equation}
	
	\begin{equation}
		\left(\int_0^N |g(x)|^p\,dx\right)^{\frac{1}{p}}
		=
		\left(\sum_{k=1}^N b_k^p\right)^{\frac{1}{p}}.
	\end{equation}
	
	\begin{equation}
		\left(\int_0^N |f(x)+g(x)|^p\,dx\right)^{\frac{1}{p}}
		=
		\left(\sum_{k=1}^N (a_k+b_k)^p\right)^{\frac{1}{p}}.
	\end{equation}
	
	Подставив $(1)$, $(2)$, $(3)$ в неравенство Гёльдера, получим:
	\[
	\sum_{k=1}^N a_k b_k
	\le
	\left(\sum_{k=1}^N a_k^p\right)^{\frac{1}{p}}
	\cdot
	\left(\sum_{k=1}^N b_k^q\right)^{\frac{1}{q}}.
	\]
	
	Подставив $(1)$, $(4)$, $(5)$ в неравенство Минковского, получим:
	\[
	\left(\sum_{k=1}^N (a_k+b_k)^p\right)^{\frac{1}{p}}
	\le
	\left(\sum_{k=1}^N a_k^p\right)^{\frac{1}{p}}
	+
	\left(\sum_{k=1}^N b_k^p\right)^{\frac{1}{p}}.
	\]
	
	\begin{remark}
		Сумма --- это интеграл подходяще выбранной функции.
	\end{remark}
\end{proof}

\begin{Theorem}{Неравенство Йенсена}{}
	Пусть $f,g$ интегрируемы на $[c,d]\subset<a,b>$ для любых $c,d$, и их несобственные интегралы сходятся.
	
	Также пусть
	\[
	g(x)\ge 0,
	\qquad
	\int_a^b g(x)\,dx = 1
	\]
	(функцию можно нормировать так, чтобы это выполнялось).
	
	Пусть $\vphi:\RR\to\RR$ --- выпуклая вниз функция. Тогда
	\[
	\vphi\left(\int_a^b g(x)f(x)\,dx\right)
	\le
	\int_a^b \vphi(f(x))g(x)\,dx.
	\]
	
	(То есть $\vphi$ от матожидания $\le$ матожидание от $\vphi$.)
	
	В частности, если $<a,b>\subset(-\infty,+\infty)$, то
	\[
	\vphi\left(\frac{1}{b-a}\int_a^b f(x)\,dx\right)
	\le
	\frac{1}{b-a}\int_a^b \vphi(f(x))\,dx.
	\]
\end{Theorem}

\begin{proof}
	На самом деле сначала докажем утверждение для сумм.
	
	Пусть
	\[
	a_k\ge 0,\qquad k=1,\ldots,N,\qquad \sum_{k=1}^N a_k>0,
	\]
	и
	\[
	-\infty < x_1 \le x_2 \le \ldots \le x_N < +\infty.
	\]
	Тогда
	\[
	\vphi\left(\frac{\sum_{k=1}^N a_kx_k}{\sum_{k=1}^N a_k}\right)
	\le
	\frac{\sum_{k=1}^N a_k\vphi(x_k)}{\sum_{k=1}^N a_k}.
	\]
	
	Положим
	\[
	\alpha_k:=\frac{a_k}{\sum_{k=1}^N a_k}.
	\]
	Тогда $\alpha_k\ge 0$ и
	\[
	\sum_{k=1}^N \alpha_k = 1.
	\]
	Следовательно, это выпуклая комбинация, и
	\[
	\vphi\left(\sum_{k=1}^N \alpha_k x_k\right)
	\le
	\sum_{k=1}^N \alpha_k \vphi(x_k).
	\]
	
	Это непосредственно следует из выпуклости: значение функции в точке выпуклой комбинации не превосходит соответствующей выпуклой комбинации значений функции.
	
	Переходя к пределу, получаем интегральную форму:
	\[
	\vphi\left(\int_a^b g(x)f(x)\,dx\right)
	\le
	\int_a^b \vphi(f(x))g(x)\,dx.
	\]
\end{proof}

\subsection{Кривые в пространстве.}

Рассмотрим вопросы:
\begin{itemize}
	\item Что такое длина кривой $\gamma$?
	\item Что такое кривая $\gamma$?
\end{itemize}

\[
\RR^d=\left\{(x_1,\ldots,x_d)\st x_k\in\RR,\ \ k=1,\ldots,d\right\},
\qquad d\in\NN.
\]

Введем метрику.

Пусть $x,y\in\RR^d$, где
\[
x=(x_1,\ldots,x_d),
\qquad
y=(y_1,\ldots,y_d).
\]
Тогда
\[
|x-y|:=\sqrt{\sum_{k=1}^d (x_k-y_k)^2}.
\]

\begin{proof}
	Проверим свойства метрики.
	\begin{itemize}
		\item $|x-y|\ge 0$, так как $\sqrt{\phantom{a}}\ge 0$.
		\item
		\[
		|y-x|
		=
		\sqrt{\sum_{k=1}^d (y_k-x_k)^2}
		=
		\sqrt{\sum_{k=1}^d (x_k-y_k)^2}
		=
		|x-y|.
		\]
		\item
		\[
		|x-y|=0
		\Longleftrightarrow
		\sum_{k=1}^d (x_k-y_k)^2=0
		\Longleftrightarrow
		(x_k-y_k)^2=0\ \ \forall k
		\Longleftrightarrow
		x_k=y_k\ \ \forall k
		\Longleftrightarrow
		x=y.
		\]
		\item Докажем неравенство треугольника:
		\[
		|x-z|\le |x-y|+|y-z|.
		\]
		Положим
		\[
		a_k:=x_k-y_k,
		\qquad
		b_k:=y_k-z_k.
		\]
		Тогда
		\[
		a_k+b_k=x_k-z_k.
		\]
		По неравенству Минковского для сумм при $p=2$ получаем
		\[
		\left(\sum_{k=1}^d (x_k-z_k)^2\right)^{\frac12}
		\le
		\left(\sum_{k=1}^d (x_k-y_k)^2\right)^{\frac12}
		+
		\left(\sum_{k=1}^d (y_k-z_k)^2\right)^{\frac12}.
		\]
		По определению это и означает:
		\[
		|x-z|\le |x-y|+|y-z|.
		\]
	\end{itemize}
\end{proof}

\begin{Definition}{Кривая в $\RR^d$}{}
	Кривая в $\RR^d$ --- множество точек $\Gamma$ в пространстве $\RR^d$:
	\[
	\Gamma=\{(\vphi_1(t),\ldots,\vphi_d(t))\}_{t\in I},
	\qquad
	I\subset\RR \ \text{--- промежуток}.
	\]
	Здесь
	\[
	\vphi_k:I\to\RR
	\]
	--- непрерывные функции.
	
	Набор функций $\vphi_k$ задает \textbf{параметризацию} точек кривой.
\end{Definition}

\begin{Definition}{Кратная точка}{}
	Точка $x\in\RR^d$ называется кратной точкой кривой, если существуют по крайней мере две различные точки $t_1,t_2\in I$ такие, что
	\[
	t_1\ne t_2,
	\qquad
	x=\vphi(t_1)=\vphi(t_2),
	\qquad
	\vphi=(\vphi_1,\ldots,\vphi_d).
	\]
\end{Definition}

\begin{Definition}{}{}
	Кривая без кратных точек (не учитывая концы) называется \textbf{простой}.
	
	Если концы совпадают, то кривая \textbf{замкнутая}.
	
	Если кратных точек конечное число, то кривая \textbf{параметризуемая}.
\end{Definition}

\begin{Definition}{Ломаная}{}
	\textbf{Ломаная} --- кривая, для которой отрезок $I=[t_0;t_N]$ \textbf{разбивается} на конечное число отрезков так, что
	\[
	\forall j \qquad \vphi_k(t), \qquad t_j\le t\le t_{j+1},
	\]
	--- линейная функция на каждом таком отрезке.
\end{Definition}

\begin{remark}
	Звенья ломаной понимаем интуитивно.
\end{remark}

\begin{Definition}{Вписанная ломаная}{}
	Пусть $\Gamma$ --- кривая в $\RR^d$.
	
	Пусть $\Gamma'$ --- ломаная, соответствующая разбиению
	\[
	\tau=\{t_0,t_1,\ldots,t_N\}
	\]
	отрезка $I$.
	
	Говорим, что $\Gamma'$ \textbf{вписана} в $\Gamma$, если
	\[
	\widetilde{\vphi}(t_j)=\vphi(t_j),
	\qquad
	0\le j\le N.
	\]
	
	\vspace{1.5\baselineskip}
	Длина ломаной $\Gamma'$ равна сумме длин ее звеньев:
	\[
	|\Gamma'|=l(\Gamma')=\Lambda_l(\Gamma').
	\]
	То есть
	\[
	l(\Gamma')
	:=
	\sum_{j=0}^{N-1}
	\sqrt{
		\sum_{k=1}^d
		\left[\widetilde{\vphi_k}(t_j)-\widetilde{\vphi_k}(t_{j+1})\right]^2
	}.
	\]
\end{Definition}

\begin{Definition}{Спрямляемая прямая}{}
	Кривая $\Gamma$ называется \textbf{спрямляемой}, если
	\[
	\sup\{l(\Gamma')\st \Gamma' \text{ --- вписанная в } \Gamma\}<+\infty.
	\]
\end{Definition}

Пример:

\[
I=(0,1],
\qquad
\vphi(t)=\left(t,\sin\frac{1}{t}\right)
\]
--- не спрямляема.

\begin{Definition}{}{}
	Если $\Gamma$ --- спрямляемая кривая в $\RR^d$, то
	\[
	l(\Gamma):=\sup\{l(\Gamma')\st \Gamma' \text{ --- ломаная, вписанная в } \Gamma\}.
	\]
\end{Definition}

\begin{Theorem}{Длина спрямляемой кривой}{}
	Пусть $\Gamma$ --- гладкая кривая в $\RR^d$, то есть
	\[
	\Gamma=\vphi(I),
	\qquad
	I=[a,b],
	\qquad
	\vphi=(\vphi_1,\ldots,\vphi_d),
	\qquad
	\vphi_n:I\to\RR,\ \ 1\le n\le d,
	\]
	и $\vphi_n'\in C([a,b])$.
	
	На концах подразумевается односторонняя производная:
	\[
	\vphi_n'(a)=\lim_{t\to a+}\frac{\vphi_n(t)-\vphi_n(a)}{t-a},
	\qquad
	\vphi_n'(b)=\lim_{t\to b-}\frac{\vphi_n(t)-\vphi_n(b)}{t-b}.
	\]
	
	Тогда $\Gamma$ спрямляема, и
	\[
	l(\Gamma)=\int_a^b \sqrt{\sum_{n=1}^d (\vphi_n'(t))^2}\,dt.
	\]
\end{Theorem}

\begin{remark}
	Важно, чтобы ломаная была \textbf{вписана} в кривую, иначе легко можно получить неверный результат или бесконечность.
	
	Вообще идеологически правильнее, выбрав точки на кривой, строить касательные в этих точках и суммировать отрезки касательных, устремляя мелкость к $0$. Но существенно проще расписать выражение для ломаной, поэтому рассмотрим именно такой подход.
\end{remark}

Далее все $\Gamma'$ считаем вписанными в кривую $\Gamma$ ломаными.

\begin{proof}
	Положим
	\[
	L:=\int_a^b \sqrt{\sum_{n=1}^d (\vphi_n'(t))^2}\,dt.
	\]
	
	Докажем два неравенства:
	\begin{enumerate}
		\item Покажем, что для любой вписанной ломаной $\Gamma'$ выполнено
		\[
		l(\Gamma')\le L.
		\]
		Тогда
		\[
		\sup\{l(\Gamma')\}=l(\Gamma)\le L.
		\]
		
		\item Явно предъявим последовательность $\{l(\Gamma_k')\}$ такую, что
		\[
		\lim_{k\to+\infty} l(\Gamma_k')=L.
		\]
		Тогда
		\[
		l(\Gamma)=\sup\{l(\Gamma')\}\ge L.
		\]
	\end{enumerate}
	Из этого получим $l(\Gamma)=L$.
	
	\vspace{2\baselineskip}
	
	\textbf{1. Оценка сверху:} $\forall\,\Gamma' \quad l(\Gamma')\le L$.
	
	Пусть $\{t_j\}_{j=0}^N$ --- точки параметризации ломаной. Тогда
	\[
	l(\Gamma')
	=
	\sum_{j=0}^{N-1}
	\sqrt{\sum_{n=1}^d \bigl(\vphi_n(t_{j+1})-\vphi_n(t_j)\bigr)^2}.
	\]
	Достаточно показать, что для каждого $j$
	\[
	\sqrt{\sum_{n=1}^d \bigl(\vphi_n(t_{j+1})-\vphi_n(t_j)\bigr)^2}
	\le
	\int_{t_j}^{t_{j+1}} \sqrt{\sum_{n=1}^d (\vphi_n'(t))^2}\,dt.
	\]
	Тогда после суммирования по $j$ получим
	\[
	l(\Gamma')
	\le
	\sum_{j=0}^{N-1}\int_{t_j}^{t_{j+1}} \sqrt{\sum_{n=1}^d (\vphi_n'(t))^2}\,dt
	=
	\int_a^b \sqrt{\sum_{n=1}^d (\vphi_n'(t))^2}\,dt
	=
	L.
	\]
	
	По формуле Ньютона--Лейбница
	\[
	\vphi_n(t_{j+1})-\vphi_n(t_j)=\int_{t_j}^{t_{j+1}} \vphi_n'(t)\,dt.
	\]
	Поэтому достаточно проверить
	\[
	\sum_{n=1}^d \left(\int_{t_j}^{t_{j+1}} \vphi_n'(t)\,dt\right)^2
	\le
	\left(\int_{t_j}^{t_{j+1}} \sqrt{\sum_{n=1}^d (\vphi_n'(t))^2}\,dt\right)^2.
	\]
	
	Положим
	\[
	f_n(t):=\bigl(\vphi_n'(t)\bigr)^2 \ge 0.
	\]
	Тогда правая часть принимает вид
	\[
	\left(\int_{t_j}^{t_{j+1}} \left(\sum_{n=1}^d f_n(t)\right)^{\frac12}\,dt\right)^2,
	\]
	а левая оценивается сверху так:
	\[
	\sum_{n=1}^d \left(\int_{t_j}^{t_{j+1}} \vphi_n'(t)\,dt\right)^2
	\le
	\sum_{n=1}^d \left(\int_{t_j}^{t_{j+1}} |\vphi_n'(t)|\,dt\right)^2
	=
	\sum_{n=1}^d \left(\int_{t_j}^{t_{j+1}} (f_n(t))^{\frac12}\,dt\right)^2.
	\]
	Теперь применяем обратное неравенство Минковского при $p=\frac12$ и получаем
	\[
	\sum_{n=1}^d \left(\int_{t_j}^{t_{j+1}} (f_n(t))^{\frac12}\,dt\right)^2
	\le
	\left(\int_{t_j}^{t_{j+1}} \left(\sum_{n=1}^d f_n(t)\right)^{\frac12}\,dt\right)^2.
	\]
	Тем самым требуемая оценка доказана.
	
	Следовательно,
	\[
	l(\Gamma')\le L
	\qquad \Longrightarrow \qquad
	l(\Gamma)=\sup\{l(\Gamma')\}\le L.
	\]
	
	\vspace{2\baselineskip}
	
	\textbf{2. Построение последовательности ломаных, длины которых стремятся к $L$.}
	
	Возьмем диадическое дробление отрезка параметризации $I=[a,b]$ ранга $k$:
	\[
	\{t_{k,j}\}_{j=0}^{N_k}.
	\]
	Построим ломаную
	\[
	\Gamma_k'=\{\vphi(t_{k,j})\}_{j=0}^{N_k}\subset \Gamma=\vphi([a,b]),
	\]
	соединяя отрезками прямой соседние точки $\vphi(t_{k,j})$ и $\vphi(t_{k,j+1})$.
	
	Тогда по построению
	\[
	l(\Gamma_k')\le l(\Gamma)\qquad \forall k\in\NN.
	\]
	
	Покажем, что
	\[
	l(\Gamma_k') \xrightarrow[k\to+\infty]{} \int_a^b \sqrt{\sum_{n=1}^d |\vphi_n'(t)|^2}\,dt = L.
	\]
	
	Так как $\vphi_n'$ непрерывны на $[a,b]$ (в концах понимаются односторонние производные), они равномерно непрерывны. Значит,
	\[
	\forall \eps>0\ \ \exists k_0\ \ \text{такое, что при } k\ge k_0
	\]
	для любого отрезка $[t_{k,j},t_{k,j+1}]$ и любых $\tau_1,\tau_2\in[t_{k,j},t_{k,j+1}]$ выполнено
	\[
	|\vphi_n'(\tau_1)-\vphi_n'(\tau_2)|<\eps,
	\qquad
	\forall n=1,\ldots,d.
	\]
	
	Сумма длин звеньев ломаной равна
	\[
	l(\Gamma_k')
	=
	\sum_{j=0}^{N_k-1}
	\sqrt{
		\sum_{n=1}^d
		\bigl(\vphi_n(t_{k,j+1})-\vphi_n(t_{k,j})\bigr)^2
	}.
	\]
	
	Оценим разность $L-l(\Gamma_k')$:
	\[
	\int_a^b \sqrt{\sum_{n=1}^d |\vphi_n'|^2(t)}\,dt - l(\Gamma_k')
	\]
	\[
	=
	\sum_{j=0}^{N_k-1}
	\int_{t_{k,j}}^{t_{k,j+1}} \sqrt{\sum_{n=1}^d |\vphi_n'|^2(t)}\,dt
	-
	\sum_{j=0}^{N_k-1}
	|t_{k,j+1}-t_{k,j}|
	\cdot
	\sqrt{
		\frac{
			\sum_{n=1}^d \bigl(\vphi_n(t_{k,j+1})-\vphi_n(t_{k,j})\bigr)^2
		}{
			|t_{k,j+1}-t_{k,j}|^2
		}
	}.
	\]
	Так как
	\[
	|t_{k,j+1}-t_{k,j}|=\int_{t_{k,j}}^{t_{k,j+1}} 1\,dt,
	\]
	можем записать
	\[
	L-l(\Gamma_k')
	=
	\sum_{j=0}^{N_k-1}
	\int_{t_{k,j}}^{t_{k,j+1}}
	\left(
	\sqrt{\sum_{n=1}^d |\vphi_n'|^2(t)}
	-
	\sqrt{
		\sum_{n=1}^d
		\frac{\bigl(\vphi_n(t_{k,j+1})-\vphi_n(t_{k,j})\bigr)^2}{|t_{k,j+1}-t_{k,j}|^2}
	}
	\right)\,dt.
	\]
	
	Для евклидовой нормы верно неравенство
	\[
	\left|
	\sqrt{\sum_{n=1}^d \alpha_n^2}
	-
	\sqrt{\sum_{n=1}^d \beta_n^2}
	\right|
	\le
	\sqrt{\sum_{n=1}^d (\alpha_n-\beta_n)^2}.
	\]
	Поэтому
	\[
	L-l(\Gamma_k')
	\le
	\sum_{j=0}^{N_k-1}
	\int_{t_{k,j}}^{t_{k,j+1}}
	\sqrt{
		\sum_{n=1}^d
		\left(
		|\vphi_n'|(t)
		-
		\frac{\vphi_n(t_{k,j+1})-\vphi_n(t_{k,j})}{|t_{k,j+1}-t_{k,j}|}
		\right)^2
	}\,dt.
	\]
	
	По теореме Коши о среднем для производных на отрезке $[t_{k,j},t_{k,j+1}]$ для каждого $n$ существует точка $\tau_{k,j}^n\in[t_{k,j},t_{k,j+1}]$ такая, что
	\[
	\frac{\vphi_n(t_{k,j+1})-\vphi_n(t_{k,j})}{|t_{k,j+1}-t_{k,j}|}
	=
	\vphi_n'(\tau_{k,j}^n).
	\]
	Поэтому
	\[
	L-l(\Gamma_k')
	\le
	\sum_{j=0}^{N_k-1}
	\int_{t_{k,j}}^{t_{k,j+1}}
	\sqrt{
		\sum_{n=1}^d
		\bigl(\vphi_n'(t)-\vphi_n'(\tau_{k,j}^n)\bigr)^2
	}\,dt.
	\]
	При $k\ge k_0(\eps)$ имеем
	\[
	|\vphi_n'(t)-\vphi_n'(\tau_{k,j}^n)|<\eps
	\qquad
	\forall n=1,\ldots,d,
	\]
	и потому
	\[
	L-l(\Gamma_k')
	\le
	\sum_{j=0}^{N_k-1}
	\int_{t_{k,j}}^{t_{k,j+1}}
	\sqrt{\sum_{n=1}^d \eps^2}\,dt
	=
	\sum_{j=0}^{N_k-1}
	\int_{t_{k,j}}^{t_{k,j+1}} \eps\sqrt{d}\,dt.
	\]
	Отсюда
	\[
	L-l(\Gamma_k')
	\le
	\eps\sqrt{d}\sum_{j=0}^{N_k-1} |t_{k,j+1}-t_{k,j}|
	=
	\eps\sqrt{d}\,(b-a).
	\]
	Итак,
	\[
	0\le L-l(\Gamma_k')\le \eps\sqrt{d}\,(b-a).
	\]
	Так как $\eps$ можно выбрать сколь угодно малым, то
	\[
	\lim_{k\to+\infty} l(\Gamma_k')=L.
	\]
	
	Следовательно,
	\[
	L\le l(\Gamma)\le L,
	\]
	а значит,
	\[
	l(\Gamma)=L=\int_a^b \sqrt{\sum_{n=1}^d |\vphi_n'(t)|^2}\,dt.
	\]
\end{proof}

\begin{Example}{Параметризация окружности}{}
	Для окружности радиуса $r>0$:
	\[
	\vphi_1(t)=r\cos(2\pi t),
	\qquad
	\vphi_2(t)=r\sin(2\pi t),
	\qquad
	t\in[0,1].
	\]
	
	$\pi$ можно понимать как абстрактное число, полученное, например, из формулы Валлиса.
	
	Тогда
	\[
	l(\Gamma)
	=
	\int_0^1
	\sqrt{
		r^2(2\pi)^2\bigl((-\sin)^2(2\pi t)+\cos^2(2\pi t)\bigr)
	}\,dt
	=
	2\pi r.
	\]
\end{Example}
