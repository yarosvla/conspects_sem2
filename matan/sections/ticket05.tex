\section{Гильбертово пространстов}

Линейное векторное с скалярным произведением, которое задает норму

\begin{Proposition}{Неравенство Коши-Буняковского-Шварца}{}

    Пусть $\mathcal{H}$~--- гильбертово, тогда
    
    \[ |<x, y>| \le ||x|| \cdot ||y|| \]
    где $x, y \in H ||x|| := \sqrt{<x, x>}$
\end{Proposition}

\begin{proof}

    $\lambda \in \RR$

    \[ 0 \le \ \ <x + \lambda y, x +\lambda y>\ \ = \ \ <x, x> + \lambda <x, y> + \lambda^2||y||^2 \]

    Дискриминант квадратного трехчлена по $\lambda \le 0$, т.е.

    \[ (<x, y>)^2 - <x, x> \cdot <y, y> \le 0 \]
\end{proof}

\[ ||x + y||^2 = \ \ <x + y, x + y> \ \ = <x, x> + <y, y> + 2<x, y> \le \]
\[ \le <x, x> + <y, y> + 2|<x,y>| \le ||x||^2 + ||y||^2 + 2||x||\cdot ||y|| \]

$l^2 \qquad ||x||_2 = \sqrt{\sum^\infty_{k=1}x^2_k}$

По КБШ:

\[
|\sum^\infty_{k=1}x_ky_k| \le \sqrt{\sum^\infty_{k=1}x^2_k}\cdot \sqrt{\sum^\infty_{k=1}y^2_k}
\]

\begin{remark}
    Ествественнее было бы рассматривать функции

    \[ h^2([a, b]) = \{f \st [a, b] \rightarrow \RR, \text{на} \ \ [a, b] \st \int^b_a|f|^2(x)dx < \infty \} \]

    \[ ||f||^2_{L^2} := \int^b_a |f|^2(x)dx \]
    
    но $\int {f}$ может стать равным 0 для $f \not = 0$

    Отложим до теории меры
\end{remark}

\begin{remark}
Сепаребельные пространства

    Далее считаем, что все наши Гельбертовы пространства~--- сепарабельные.

    То есть существует счётное всюду плотное подмножество. (подмножество, замыкание которого равно всему множеству, но для этого нужна топология)

    Т.е. $\forall \ \ \mathcal{H} \ \ \exists \ \ E \subset \mathcal{H}$ (счетное) такое что $\forall \ \ x \in \mathcal{H} \exists \{ x_k \} \subset E \st ||x_k - x||_{\mathcal{H}} \longrightarrow 0 (n \rightarrow +\infty)$
\end{remark}

\begin{Definition}{}{}
    $\mathcal{H}$~--- Гильбертово пространство, ${x_k} \subset H$

    Такое что, $x \longrightarrow x \Leftrightarrow ||x_k - x||_{\mathcal{H}} \rightarrow 0 (n \rightarrow +\infty)$

    Ряд $\sum^{\infty}_{k=1}x_k$ сходится в $\mathcal{H}$ к $S \in \mathcal{H}$, елси

    \[ ||\sum^N_{k=1}x_k - S|| \longrightarrow 0 (N \rightarrow +\infty) \]
\end{Definition}

\begin{Definition}{}{}
    $\mathcal{H}$~--- Гильбертово, $x, y \in \mathcal{H} \st x \perp y \Leftrightarrow <x, y> = 0 $

    Система \textbf{ненулевых} векторов $\{x_k\}_{\mathcal{H}}$ является ортогональной, если $x_{\alpha} \perp x_{\beta} \ \ \forall \ \ \beta \not = \alpha $
\end{Definition}

\begin{Proposition}{}{}
    Вектора в ортогнальной системе линейно независимы, т.е. если $\sum^N_{k=1}\lambda_k x_k = 0$, где $x_k \in \{x_\alpha\}_{\alpha \in \mathcal{H}}$~--- система, то $l_k \lambda_k = 0$
\end{Proposition}

\begin{proof}
    $0 = \sum \lambda_k x_k = <>$
\end{proof}

\begin{Definition}{}{}
    Какая-то система $\{x_\alpha\}_{\alpha \in \mathcal{H}}$~--- полная, если 

    \[ clos span {x_alpha} = \mathcal{H} \]
    т.е. 
    \[ clos \{ \sum^N_{k+1}\lambda_kx_k \st \lambda_k \in \RR, x_\alpha \in \{ x_\alpha\}_{\alpha \in \mathcal{H}} \} \]

\end{Definition}

    \[ g_k \perp g_j; j \not = k \rightarrow \int^1_0 \cos{2\pi kx} \cdot \cos{2\pi jx}dx = 0 \]

\begin{Proposition}{}{}
    $\mathcal{H} \ \ $ ~--- сепарабельное Гильбертово $\Rightarrow \{ x_\alpha\}_{\alpha \in \mathcal{H}}$~--- Ортогональная система не более чем счетная 
\end{Proposition}

\begin{proof}
    
    WLOG $\{ x_\alpha\}_{\alpha \in \mathcal{H}}$~--- ортонормированная система

    Рассмотрим норму разности:

    \[ ||x_\alpha - x_\beta|| = \sqrt{<x_\alpha, x_\alpha> - 2<x_\alpha, x_\beta> + <x_\beta, x_\beta>} = \sqrt{2} \]

    Рассмотрим стсему шаров $\{ B_\alpha \} \st$

    \[ B_\alpha = \{ x \st ||x - x_\alpha|| < \frac{1}{2} \}\]

    $B_\alpha \cap B_\beta \not = \emptyset, \qquad \alpha \not = \beta$

    Т.к. $\ \ \mathcal{H} \ \ $~--- сепарабельная $\Rightarrow \ \ \exists \ \ E = \{ y_k \}_{k \in \NN} \st clos E = \ \ \mathcal{H}$

    Очевидно, что $\ \ \forall \alpha B_\alpha \ni y_{n(\alpha)}$ для какого-то $n(\alpha) \Rightarrow$
    
    Такая $B_\alpha$ тоже не более, чем счетна
\end{proof}

\begin{Theorem}{}{}
    Пусть $\ \ \mathcal{H}$~--- сем Гильбертово пространство

    $\{ f_k\}_{k \in \NN}$~--- какая-то линейно независимая система векторов в $\ \ \mathcal{H}, f_k \not = 0$

    Тогда $\ \ \exists \ \ \{ \vphi_k\}_{k \in \NN}$

    \begin{enumerate}
        \item $\{ \vphi_k\}$~--- ортонормированная система в $\ \ \mathcal{H}$
        \item $\vphi_k = \sum_{j=1}^k \lambda_{jk} f_j$~--- линейная комбинация первых $k$ векторов из $\{ \vphi_k\}$
        \item $f_k = \sum _{j=1}^k \tilde{\lambda_{jk}} \vphi_j$~--- Линейная комбинация первых $k$ векторов из ...
    \end{enumerate}
\end{Theorem}

\begin{proof}
    
    База: 
    \[ \vphi_1 := \frac{f_1}{||k||}\]
    Переход:
    
    Пусть $\vphi_1, \ldots, \vphi_{k-1}$ удовлетворяют $1-3$

    \textbf{Хотим:} каждый $\vphi_k$ из $f_1, \ldots, f_{k-1}, f_k$, так, чтобы выполнились $1-3$

    3): означает, что 
    \[ f_k = \sum_{j=1}^k\tilde{\lambda}_{jk} \vphi_j = \tilde{\lambda}_{1k}\cdot \vphi_1 + \ldots + \tilde{\lambda}_{(k-1), k}\cdot \vphi_{(k-1)} + g_k \]

    Если бы такое представление было, то $<f_k, \vphi_j> = \tilde{\lambda}_{1k}<\vphi_1, \vphi_j> + \ldots + \tilde{\lambda}_{j, k}<\vphi_i, \vphi_i> + \ldots + \tilde{\lambda}_{(k-1), k}\cdot <\vphi_{(k-1)}, \vphi_j + <g_k, \vphi_j =$

    \[ 0 + 0 + \ldots + \tilde{\lambda}_{jk} ||\vphi_i||^2 + 0 + \ldots + (<g_k, \vphi_i> = 0) \]

    $<g_k, \vphi_i> = 0$, т.к. ожидаем, что $\vphi_k \perp \vphi_i$
    
    Итак, если прадставление есть, то

    \[ \tilde{\lambda}_{jk} = <f_k, \vphi_i>, \qquad j < k \]

    Положим: 

    \[ g_k := f_k - \sum_{j=1}^{k-1}\tilde{\lambda}_{j, k}\cdot \vphi_j \]

    \[ \vphi_k := \frac{g_k}{||g_k||} \]

    При этом $g_k \not = 0, j \not = 0 , \{ f_k\}$~--- линейно независимые

    $\tilde{\lambda}_{k, k} := ||g_k|| \not = 0.$ Кандидаты на коэффициенты есть. Проверим 1-3.

    3. $f_k = \sum_{j=1}^k \tilde{\lambda}_{j,k} \cdot \vphi_k$~--- прямо по определению
    1. $\{ \vphi_1, \ldots, \vphi_k \}$ ортогональная система
     $\{ \vphi_1, \ldots, \vphi_{k-1} \}$~--- ортонормировання система, $\ \ ||\vphi_k|| = 1$ по построению

     $<\vphi_k, \vphi_j> $
     ...

    2. $\vphi_k = \sum^k_{j=1} \lambda_{j, k} f_j$~--- также очевидно, т.к.

    \[ \vphi_k = \frac{f_k - \sum_{j=1}{k-1}\tilde{\lambda}_{j, k-1} \vphi_j}{||f_k - \sum_{j=1}^{k-1}\tilde{\lambda}_{j, k-1}\cdot \vphi_j||} = \ \ \text{линейная комбинация из} \ \ f_i , \ \ 1 \le i \le K\]
\end{proof}

\textbf{Далее все Гильбертовы пространства будут полными}

Хочется проверить, что если

$\{ e_k \}_{k \in \NN}$~--- базис в $\ \ \mathcal{H} \ \ $, то $\ \ \forall x \in \mathcal{H}$~ есть разложение:

\[ x = \sum_{k=1}^\infty c_k e_k, \qquad c_k \in \RR \]

$<x, e_j> = \sum_{k=1}^\infty c_k <e_k, e_j> = ...$

\begin{Definition}{}{}
    Пусть $\{ e_k\}$~--- ортонормировання система в $\ \ \mathcal{H}, \qquad x \in \mathcal{H}$
    
    $c_k = <x, e_k>$~--- коэффициент Фурье элемента $x$, по системе $\{ e_k\}_{k \in \NN}$

    Ряд (формальный) $\sum_{k=1}^\infty$~--- ряд Фурье элемента $x$.

    
\end{Definition}

Хочется, чтобы $\sum_{k=1}^\infty c_k e_k = x, если \{ e_k \}$~--- ортонормированный базис в $\ \ \mathcal{H}$text

\begin{proof}
    $N$ элементов~--- $e_1, \ldots, e_N$. Рассмотрим линейную комбинацию:

    \[ \sum_{k=1}^\infty \lambda_k \cdot e_k, \qquad \lambda_1, \ldots, \lambda_k \in \RR \]

    Минимальное расстояние:

    \[ ||x - \sum_{k=1}^N \lambda_k e_k || = dist_{\mathcal{H}}(x, span\{ e_k\}_{k=1}^N) \]

    \[ ||x - \sum_{k=1}^N \lambda_k e_k ||^2 = <x - \sum_{k=1}^N \lambda_k e_k, x - \sum_{j=1}^N \lambda_j e_j> = \]

    \[ = <x, x> - 2<x, \sum{k=1}^N\lambda_ke_K> + <\sum_{k=1}^N \lambda_k e_k, \sum_{j=1}^N \lambda_j e_j = \]

    \[ = ||x||^2 - 2\sum_{k=1}^N \lambda_k c_k + \sum_{k, j=1}^N\lambda_k c_j .... \]

    \[ = ||x||^2 - 2 \sum_{k=1}^N\lambda_k c_k + \sum_{1}^N \lambda_k^2 \cdot 1 = ... \]
    
    \[ ||x||^2 - \sum_{k=1}^Nc_k^2 + () = \]
    \[ = ||x||^2 - \sum_{k=1}^N ... \]

    Минимум: $\lambda_j = c_j, \qquad 1 \le j \le N$, т.е.

    \[ dist^2(x, span\{e_k\}_{K=1}^N) = ||x||^2 - \sum_{k=1}^Nc_k^2 \ge 0 \ \ \forall \ \ N \]
    ...
...
    Неравенство Бесселя: $\sum_{k=1}^\infty c_k^2 \le ||x||^2$
\end{proof}

Пусть теперь $\{ e_k\}_{k \in \NN}$~--- ортонормированный базис, т.е. $clos span \{ e_k \} = \mathcal{H} \Leftrightarrow$

\[ \forall \ \ x \in \mathcal{H} \qquad \exists \ \ \{\lambda_k\} \st \sum_{k=1}^N \lambda_k e_k \rightarrow x (N \rightarrow \infty) \]

т.е.:

\[ || x - \sum_{k1}^N \lambda_k e_K||^2 \longrightarrow 0 \]

\[ ||x||^2\cdot \sum_{k=1}^N c_k^2 \longrightarrow 0 (n \rightarrow \infty) \]

\begin{Proposition}{}{}
    Если $\{ e_k\}_{k \in \NN}$~--- Ортонормированный базис, то верно равенство Парсеваля:

    \[ \sum_{k=1}^\infty c_K^2 = ||x||^2, \qquad c_k = <x_k, x_k> \]
\end{Proposition}


Итак: $\ \ \mathcal{H}$~--- сем .. Гильбертово пространство; $\{ e_k \}$~--- ортонормированный базис в $\mathcal{H}$

\[ \mathcal{H} \ni x \]


\begin{Theorem}{Рисс-Фишер}{}
    Пусть $\{ e_k\}$~--- ортонормированный базис в $\mathcal{H} \st c = (c_1, \ldots, c_k, \ldots) \in l^2 \Rightarrow$

    $\exists ! \ \ x \in \mathcal{H} \st c_k = <x_1, e_k>_\mathcal{H}$ и

    $||c||_{l^2} = ||x||_{\mathcal{H}}$
\end{Theorem}

\begin{proof}

    Конечно же $x = \sum_{k=1}^\infty c_k e_k$

    Проверим, что ряд сходится, т.е. последовательность частичных сумм куда-то сходятся

    Проверим, что это последовательность Коши, т.е. 
    
    \[ \forall \ \ \eps > 0 \ \ \exists \ \ N(c) \st \forall \ \ n \ge N(\eps), \forall \ \ m \in \NN \]

    ...

    \[ ||S_{N + m} - S_N||^2 = ||\sum_{k = N+1}^{N+m} c_k e_k|| \]

    но $c = (c_1, \ldots, c_k, \ldots)$
    ...

    т.е. ряд $\sum_{k=1}^\infty c_k e_k$ сходится, обозначим эту сумму ща $x$


    Осталось проверить, что $<x, e_k> = c_k$ и единственность

    \[ <x, e_k> \overset{def}{=} <x - S_N, e_k>_{\mathcal{H}} + <S_N, e_k>_{\mathcal{H}} \qquad \forall \ \ N \]

    \[ <S_N, e_k> = <\sum_{j=1}^N c_je_j, e_k> = c_k \]

    \[ |<x - S_N, e_k>| \le ||f - S_N|| \cdot ||e_k|| \longrightarrow 0 (N \rightarrow +\infty) \]

    Единственность:

    Пусть $<x, y> = <y, e_k> = c_k \qquad \forall \ \ k \Rightarrow$

    $(x - y) \perp e_k \ \ \forall \ \ k \Rightarrow (x - y) = 0$ т.к. $\{e_k\}$~--- ортонормированный базис

    Пусть $(x-y) = z \not = 0 \Rightarrow c'_k := <z, e_k> \not = 0$ (не равняется тождественно)

    \[ dist^2(z, span \{ e_k\}_{k=1}^N) = ||z||^2 - \sum_{k=1}^N c'_k^2 \]

    С другой стороны, $||z - \sum_{k=1}^N c'_k e_k|| \longrightarrow 0 (N \rightarrow 0) (?!)$
\end{proof}

\textbf{Следствие:}

    Любое Гильбертово пространство $\mathcal{H} \ \ $ изоморфно $l^2$, т.е. есть биекция $\mathcal{A} \st \mathcal{H} \rightarrow e^2$:

    \[ ||\mathcal{A}x||_{l^2} = ||x||_{l^2} \]

    $\mathcal{A}$~--- линейный оператор:

    $<\mathcal{A}x, \mathcal{A}y>_{l^2} = <x, y>_{\mathcal{H}}$

\begin{proof}

    Проверим линейность оператора $\mathcal{A}$:

    В $\mathcal{H}$~ есть ортонормированный базис $\{ e_k\}_{k+1}^\infty \st \mathcal{A}x = c = \{ c_k \}$
    
    $<\mathcal{A}x, \mathcal{A}y>_{l^2} \overset{\text{def}}{=}$

    ...

    \[ ||<x - S_N, y - \tilde{S_N}|| \le ||x - S_N||\cdot ||y - \tilde{S_N}|| \longrightarrow 0 (n \rightarrow \infty) \]

    \[ |<x - S_N, \tilde{S_N}| \longrightarrow 0 (n \rightarrow \infty) \]

    \[ |<S_N, y - \tilde{S_N}| \longrightarrow 0 (n \rightarrow \infty) \]

    \[ <S_N, \tilde{S_N}> = <\sum_{k=1}^Nc_n e_k, \sum_{j=1}^N d_j e_j> = \sum_{k=1}^N c_k d_k \]

    \[ т.е. |<x, y>_{\mathcal{H}} - \sum_{k=1}^\infty c_k d_k| \longrightarrow 0 (N \rightarrow \infty) \]

    <x, y> = \sum_{k=1}^\infty c_k d_k
\end{proof}

Recall that:

X~--- Банахово пространство, если X~--- линейное нормированное пространство, полное относительно метрики, порождаемой своей нормой

\sum_{k=1}^\infty \lambda_k x_k ~--- сходится в X, если \ \ \exists \ \ x \in X \st || x - \sum_{k=1}^N\lambda_k x_k||_X \longrightarrow 0 (N \rightarrow \infty)

\begin{Definition}{}{}
    Пусть $X, Y$~--- линейные векторные пространства.

    Отображение $\mathcal{A} \st X \rightarrow Y$~--- \textbf{линейный оператор}, если оно линейное:

    \[ \mathcal{A}(x + \lambda y) = \mathcal{A}x + \lambda \mathcal{A}y, \qquad \forall \ \ x, y \in X, \lambda \in \RR \]

    Если $Y = \RR$, то линейный оператор называется \textbf{линейным функционалом}.
\end{Definition}

\begin{Definition}{}{}
    Пусть $X, Y$~--- Банаховы. $\qquad \mathcal{A} \st X \rightarrow Y$~--- линейный оператор

    \mathcal{A} \st Dom \mathcal{A} \rightarrow Range \mathcal{A}, где Dom \mathcal{A}~--- линейное пространство X, а Range \mathcal{A}~--- линейное подпространство Y, т.е.
    
    если x, y \in Dom \mathcal{A} \Rightarrow x + \lambda y \in Dom \mathcal{A}, \ \ \forall \ \ \lambda \in \RR. Аналогично для Range. 
\end{Definition}

\begin{Example}{}{}
    X = C([0, 1])

    ||f||_X := sup_{x \in [0, 1]} |f(x)|

    \mathcal{A}f := f' \qquad \mathcal{A} = \{ f \in C([0, 1]) \st \exists \exists f(x) \ \ \forall \ \ x \in (0, 1] ) \}
\end{Example}

Зададим норму оператора $\mathcal{A}$:

\[ ||\mathcal{A} := \sup_{||x|| \le 1} ||\mathcal{A}x||_Y \]

Оператор называется ограниченным, если $||\mathcal{A}|| < +\infty$

\begin{Example}{}{}
    $(x^k)' = kx^{k-1} = \RR$, при $x = 1$

    ||x^k||_{C([0, 1])} = max_{0 \le x \le 1} |x^k| = 1

    Пример неограниченного оператора:

    $(\mathcal{A}f)(x) := f(x)g(x), \qquad \text{где} \ \ g \in C([0,1])$~--- фиксирована

    \[ ||\mathcal{A}|| := \sup_{||f||_{C([0,1])}} ||\mathcal{A}f|| = \sup_{x \in [0, 1], f \in C([0,1 ])} |f(x)g(x)| \le \]

    \[ \sup |f(x)| \cdot \sup |g(x)| < +\infty \]
\end{Example}

\begin{Proposition}{}{}
    Пусть $\mathcal{A} \st X \rightarrow Y$~--- линейный, $\qquad X, Y$~--- Банаховы пространства, тогда

    \[ ||\mathcal{A}|| = \sup_{||x||_X = 1}||\mathcal{Ax}||_Y = \sup_{x \in X, x \not = 0} \frac{||\mathcal{A}x||}{||x||} \]
\end{Proposition}

\begin{proof}

    \[ ||\mathcal{A}|| \ge \sup_{||x||_X = 1} ||\mathcal{A}x||_Y \qquad (\text{на самом деле равенство})\]

    \[ \forall x \not = 0 \st ||\mathcal{A}x||_Y = ||\mathcal{A}(\frac{x}{||x||_X})||_Y\cdot ||x||_X \]

    \[ x = 0 \Rightarrow ||\mathcal{A}x||_Y = 0 \]
\end{proof}

\begin{Proposition}{}{}

    $X, Y$~--- Банаховы пространства.
    \mathchar{A} \st X \rightarrow Y~--- ограничен \Leftrightarrow \mathcal{A}~--- непрерывный как отображение \Leftrightarrow непрерывный в нуле

\end{Proposition}

\begin{proof}

    \begin{enumerate}
        \item Непрерывность в нуле $\Rightarrow$ непрерывность
        
        Неравенство в нуле:
        \[ \forall \ \ \eps > 0 \ \ \exists \ \ \delta > 0 \qquad ||x||_X < \delta \Rightarrow ||\mathcal{A}x||_Y < \eps \]

        \item Непрерывность в точке $Y \in Dom \mathcal{A}$
        
        \[ \forall \ \ \eps > 0 \ \ \exists \ \ \delta > 0 \qquad ||y-x||_X < \delta \Rightarrow ||\mathcal{A}y - \mathcal{A}x|| = ||\mathcal{A}(y-x)|| < \eps \]

        \item Непрерывность $\Rightarrow$ ограниченность
        
        Покажем, что если $\mathcal{A}$~--- непрерывный, то $||\mathcal{A}|| < + \infty$

        Рассмотрим \sup_{||x||_X \le 1} ||\mathcal{A}x||_Y

        Непрерывность в нуле: 

        \[ \forall \ \ \eps > 0 \ \ \exists \ \ \delta > 0 \st ||x||_X \le \delta \Rightarrow ||\mathcal{A}x||_Y \le \eps \]

        \[ \eps := 1 \qquad ||\mathcal{A}x|| = \frac{1}{\delta}||\mathcal{A}(\frac{1}{\delta})||\]

        \[ \forall \ \ x' \st ||x'||_X \le 1 ||\delta x'||_x \le \delta \Rightarrow ||\mathcal{A}x'||_Y = \frac{1}{\delta}||\mathcal{A}(x'\delta)||_Y \le \frac{1}{\delta} \]
