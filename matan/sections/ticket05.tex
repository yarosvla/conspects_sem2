\section{Гильбертово пространство}

\begin{Definition}{Гильбертово пространство}{}

    Гильбертово пространство $\ \ \mathcal{H} \ \ $~--- линейное нормированное пространство, на котором задано скалярное произвдение

    $\langle \cdot, \cdot \rangle \st V \times V \longrightarrow \RR$

    Со свойствами:

    \begin{enumerate}
        \item $\langle v, v \rangle \ge 0, \qquad \langle v, v \rangle = 0 \Leftrightarrow v = 0$
        \item $\langle \lambda v_1, v_2 \rangle = \lambda \langle v_1, v_2 \rangle = \langle v_1, \lambda v_2 \rangle $
        \item $\langle v_1 + v_2, v_3 \rangle = \langle v_1, v_2 \rangle + \langle v_2, v_3 \rangle$
        \item $\langle v_1, v_2 \rangle = \langle v_2, v_1 \rangle$
    \end{enumerate}
\end{Definition}

Гильбертовы пространства автоматически нормированны.

\begin{Proposition}{Неравенство Коши-Буняковского-Шварца}{}

    Пусть $\mathcal{H}$~--- гильбертово, тогда
    
    \[ |\langle x, y \rangle| \le  \| x \|  \cdot  \| y \|  \]
    где $x, y \in H, |\qquad |x \|  := \sqrt{\langle x, x \rangle }$
\end{Proposition}

\begin{proof}

    $\lambda \in \RR$

    \[ 
    0 \le \ \ \langle x + \lambda y, x + 2\lambda y \rangle \ \ = \ \  \langle x, x \rangle  + \lambda  \langle x, y \rangle  + \lambda^2 \| y \| ^2 
    \]

    Дискриминант квадратного трехчлена по $\lambda \le 0$, т.е.

    \[ ( \langle x, y \rangle )^2 -  \langle x, x \rangle  \cdot  \langle y, y \rangle  \le 0 \]
\end{proof}

\[  \| x + y \| ^2 = \ \  \langle x + y, x + y \rangle  \ \ =  \langle x, x \rangle  +  \langle y, y \rangle  + 2 \langle x, y \rangle  \le \]
\[ \le  \langle x, x \rangle  +  \langle y, y \rangle  + 2| \langle x,y \rangle | \le  \| x \| ^2 +  \| y \| ^2 + 2 \| x \| \cdot  \| y \|  \]

$l^2 \qquad  \| x \| _2 = \sqrt{\sum^\infty_{k=1}x^2_k}$

По КБШ:

\[
|\sum^\infty_{k=1}x_ky_k| \le \sqrt{\sum^\infty_{k=1}x^2_k}\cdot \sqrt{\sum^\infty_{k=1}y^2_k}
\]

\begin{remark}

    Ествественнее было бы рассматривать функции

    \[ h^2([a, b]) = \{f \st [a, b] \rightarrow \RR, \text{на} \ \ [a, b] \st \int^b_a|f|^2(x)dx  \langle  \infty \} \]

    \[  \| f \| ^2_{L^2} := \int^b_a |f|^2(x)dx \]
    
    но $\int {f}$ может стать равным 0 для $f \not = 0$

    Отложим до теории меры
\end{remark}

\begin{remark}

    Далее считаем, что все наши Гельбертовы пространства~--- сепарабельные.

    То есть существует счётное всюду плотное подмножество. (подмножество, замыкание которого равно всему множеству, но для этого нужна топология)

    Т.е. $\forall \ \ \mathcal{H} \ \ \exists \ \ E \subset \mathcal{H}$ (счетное) такое что $\forall \ \ x \in \mathcal{H} \ \ \exists \ \ \{ x_k \} \subset E \st  \| x_k - x \| _{\mathcal{H}} \longrightarrow 0 (n \rightarrow +\infty)$
\end{remark}

\begin{Definition}{Сходимость ряда в гильбертовом пространстве}{}
    $\mathcal{H}$~--- Гильбертово пространство, ${x_k} \subset H$

    Такое что, $x_n \longrightarrow x \Leftrightarrow  \| x_n - x \| _{\mathcal{H}} \longrightarrow 0 \quad (n \rightarrow +\infty)$

    Ряд $\sum^{\infty}_{k=1}x_k$ сходится в $\mathcal{H}$ к $S \in \mathcal{H}$, если

    \[  \| \sum^N_{k=1}x_k - S \|  \longrightarrow 0 \quad (N \rightarrow +\infty) \]
\end{Definition}

\begin{Definition}{}{}
    $\mathcal{H}$~--- Гильбертово, $x, y \in \mathcal{H} \st x \perp y \Leftrightarrow  \langle x, y \rangle  = 0 $

    Система \textbf{ненулевых} векторов $\{x_k\}_{\mathcal{H}}$ является \textbf{ортогональной}, если $x_{\alpha} \perp x_{\beta} \ \ \forall \ \ \beta \not = \alpha $
\end{Definition}

\begin{Proposition}{}{}
    Векторы в ортогнальной системе линейно независимы, т.е. если $\sum^N_{k=1}\lambda_k x_k = 0$, где $x_k \in \{x_\alpha\}_{\alpha \in \mathcal{H}}$~--- система, то $l_k \lambda_k = 0$
\end{Proposition}

\begin{proof}
    \[
    0 = \sum \lambda_k x_k =  \langle \sum_{k=1}^N \lambda_k x_k, \; x_{k0} \rangle =
    \]
    \[
    \sum_{k=1}^N \lambda_k \langle x_k, x_{k0} \rangle = \lambda_{k0} \langle x_{k0}, x_{k0} \rangle \Rightarrow \lambda_{k0} = 0
    \]
\end{proof}

\begin{Definition}{}{}
    Система $\{x_\alpha\}_{\alpha \in \mathcal{H}}$~--- полная, если 

    \[ clos \; span \{x_\alpha\} = \mathcal{H} \]
    т.е. 
    \[ 
    clos \{ \sum^N_{k+1}\lambda_kx_k \st \lambda_k \in \RR, x_\alpha \in \{ x_\alpha\}_{\alpha \in \mathcal{H}} \} = \ \ \mathcal{H}
    \]

    Полная ортогональная система~--- $\textbf{базис}$ в \ \ $\mathcal{H}$
\end{Definition}

\begin{Example}{}{}
    Базис в $ l^2 \st x_k = (0, \ldots, 0, 1, 0, \ldots)$
\end{Example}

\begin{Definition}{}{}
    Базис ортонормированный, если $ \| x_\alpha \|  = 1 \ \ \forall \alpha$
\end{Definition}

\begin{Example}{}{}
    Рассмотрим базис и на него натянем гильбертово пространство

    Рассмотрим пространство непрерывных функций на отрезке $[0, 1] - C([0, 1])$

    $ \langle f, g \rangle  = \int^1_0 f(x)g(x)dx$

    $f_0(x) := 0, \qquad f_k(x) := \sin{2\pi kx}, \qquad k \in \NN, \ \ x \in [0, 1]$

    $g_0(x) := 1, \qquad g_k(x) := \cos{2\pi kx}, \qquad k \in \NN, \ \ x \in [0, 1]$

    \[ f_k \perp f_j; j \not = k \rightarrow \int^1_0 \sin{2\pi kx} \cdot \sin{2\pi jx}dx = 0 \]
    \[ g_k \perp g_j; j \not = k \rightarrow \int^1_0 \cos{2\pi kx} \cdot \cos{2\pi jx}dx = 0 \] 
    \[ f_k \perp g_j; \forall \ \ j, k \rightarrow \int^1_0 \sin{2\pi kx} \cdot \cos{2\pi jx}dx = 0  \ \ \forall k, j \]

    \[ \int_{0}^{1} \sin^2{2\pi kx} dx = \frac{1}{2} \]

    \[ clos \ \ span \{ f_k, g_k \} \supset C([0, 1]) \]
\end{Example}

\begin{Proposition}{}{}
    $\mathcal{H} \ \ $ ~--- сепарабельное гильбертово $\Rightarrow \{ x_\alpha\}_{\alpha \in \mathcal{H}}$~--- ортогональная система, не более чем счетная 
\end{Proposition}

\begin{proof}
    
    WLOG $\{ x_\alpha\}_{\alpha \in \mathcal{H}}$~--- ортонормированная система

    Рассмотрим норму разности:

    \[  \| x_\alpha - x_\beta \|  = \sqrt{ \langle x_\alpha, x_\alpha \rangle  - 2 \langle x_\alpha, x_\beta \rangle  +  \langle x_\beta, x_\beta \rangle } = \sqrt{2} \]

    Рассмотрим стсему шаров $\{ B_\alpha \} \st$

    \[ B_\alpha = \{ x \st  \| x - x_\alpha \|   <  \frac{1}{2} \}\]

    $B_\alpha \cap B_\beta \not = \varnothing, \qquad \alpha \not = \beta$

    Т.к. $\ \ \mathcal{H} \ \ $~--- сепарабельная $\Rightarrow \ \ \exists \ \ E = \{ y_k \}_{k \in \NN} \st clos E = \ \ \mathcal{H}$

    Очевидно, что $\ \ \forall \alpha \ \ B_\alpha \ni y_{n(\alpha)}$ для какого-то $n(\alpha) \Rightarrow$
    
    Такая $B_\alpha$ тоже не более, чем счетна
\end{proof}

\begin{Theorem}{}{}
    Пусть $\ \ \mathcal{H}$~--- сепарабельное гильбертово пространство

    $\{ f_k\}_{k \in \NN}$~--- какая-то линейно независимая система векторов в $\ \ \mathcal{H}, \ \ f_k \not = 0$

    Тогда $\ \ \exists \ \ \{ \vphi_k\}_{k \in \NN}$

    \begin{enumerate}
        \item $\{ \vphi_k\}$~--- ортонормированная система в $\ \ \mathcal{H}$
        \item $\vphi_k = \sum_{j=1}^k \lambda_{jk} f_j$~--- линейная комбинация первых $k$ векторов из $\{ f_k\}$
        \item $f_k = \sum _{j=1}^k \tilde{\lambda}_{jk} \vphi_j$~--- линейная комбинация первых $k$ векторов из $\{ \vphi_k \}$
    \end{enumerate}
\end{Theorem}

\begin{proof}
    
    База: 
    $ \vphi_1 := \frac{f_1}{ \| k \| } $
    Переход:
    
    Пусть $\vphi_1, \ldots, \vphi_{k-1}$ удовлетворяют $1-3$

    \textbf{Хотим:} получить $\vphi_k$ из $f_1, \ldots, f_{k-1} \Rightarrow f_k$, так, чтобы выполнились $1-3$

    3): означает, что 
    \[ f_k = \sum_{j=1}^k\tilde{\lambda}_{jk} \vphi_j = \tilde{\lambda}_{1k}\cdot \vphi_1 + \ldots + \tilde{\lambda}_{(k-1), k}\cdot \vphi_{(k-1)} + g_k \]

    Если бы такое представление было, то 
    \[
    \langle f_k, \vphi_j \rangle  = \tilde{\lambda}_{1k} \langle \vphi_1, \vphi_j \rangle  + \ldots + \tilde{\lambda}_{j, k} \langle \vphi_i, \vphi_i \rangle  + \ldots + \tilde{\lambda}_{(k-1), k}\cdot  \langle \vphi_{(k-1)}, \vphi_j \rangle +  \langle g_k, \vphi_j \rangle =
    \]

    \[
    = 0 + 0 + \ldots + \tilde{\lambda}_{jk}  \| \vphi_i \| ^2 + 0 + \ldots + ( \langle g_k, \vphi_i \rangle  = 0) 
    \]

    $ \langle g_k, \vphi_i \rangle  = 0$, т.к. ожидаем, что $\vphi_k \perp \vphi_i$
    
    Итак, если представление есть, то

    \[ \tilde{\lambda}_{jk} =  \langle f_k, \vphi_i \rangle , \qquad j  <  k \]

    Положим: 

    \[ g_k := f_k - \sum_{j=1}^{k-1}\tilde{\lambda}_{j, k}\cdot \vphi_j \]

    \[ \vphi_k := \frac{g_k}{ \| g_k \| } \]

    При этом $g_k \not = 0, j \not = 0 , \{ f_k\}$~--- линейно независимые

    $\tilde{\lambda}_{k, k} :=  \| g_k \|  \not = 0$.
    
    Кандидаты на коэффициенты есть. Проверим 1-3.

    3. $f_k = \sum_{j=1}^k \tilde{\lambda}_{j,k} \cdot \vphi_k$~--- прямо по определению

    1. $\{ \vphi_1, \ldots, \vphi_k \}$ ортогональная система

    $\{ \vphi_1, \ldots, \vphi_{k-1} \}$~--- ортонормировання система, $\ \  \| \vphi_k \|  = 1$ по построению

    $ \langle \vphi_k, \vphi_j \rangle, j_0 < k $
    
    \[
    \langle g_k, \vphi_{j0} \rangle = \langle f_k - \sum_{j=1}^{k-1} \tilde{\lambda}_{jk} \cdot \vphi_j, \ \ \vphi_{j0}, \rangle =
    \]
    \[
    = \langle f_k, \vphi_{j0} \rangle - \tilde{\lambda}_{j_0k} \langle \vphi_{j0}, \vphi_{j0} \rangle = 0
    \]

    2. $\vphi_k = \sum^k_{j=1} \lambda_{j, k} f_j$~--- также очевидно, т.к.

    \[ \vphi_k = \frac{f_k - \sum_{j=1}{k-1}\tilde{\lambda}_{j, k-1} \vphi_j}{ \| f_k - \sum_{j=1}^{k-1}\tilde{\lambda}_{j, k-1}\cdot \vphi_j \| } = \ \ \text{линейная комбинация из} \ \ f_i , \ \ 1 \le i \le K\]
\end{proof}

\textbf{Далее все Гильбертовы пространства будут полными}

Хочется проверить, что если

$\{ e_k \}_{k \in \NN}$~--- базис в $\ \ \mathcal{H} \ \ $, то $\ \ \forall x \in \mathcal{H}$~ есть разложение:

\[ x = \sum_{k=1}^\infty c_k e_k, \qquad c_k \in \RR \]

$ \langle x, e_j \rangle  = \sum_{k=1}^\infty c_k  \langle e_k, e_j \rangle  = c_j \cdot \langle e_j, e_j \rangle = c_j$, если $\{ e_k \}$~--- ОНС

\begin{Definition}{}{}
    Пусть $\{ e_k\}$~--- ортонормировання система в $\ \ \mathcal{H}, \qquad x \in \mathcal{H}$
    
    $c_k =  \langle x, e_k \rangle $~--- коэффициент Фурье элемента $x$, по системе $\{ e_k\}_{k \in \NN}$

    Ряд (формальный) $\sum_{k=1}^\infty$~--- ряд Фурье элемента $x$.

    
\end{Definition}

Хочется, чтобы $\sum_{k=1}^\infty c_k e_k = x, \text{если} \ \ \{ e_k \}$~--- ортонормированный базис в $\ \ \mathcal{H}$

\begin{proof}
    $N$ элементов~--- $e_1, \ldots, e_N$. Рассмотрим линейную комбинацию:

    \[ \sum_{k=1}^\infty \lambda_k \cdot e_k, \qquad \lambda_1, \ldots, \lambda_k \in \RR \]

    Минимальное расстояние:

    \[  \| x - \sum_{k=1}^N \lambda_k e_k  \|  = dist_{\mathcal{H}}(x, span\{ e_k\}_{k=1}^N) \]

    \[  \| x - \sum_{k=1}^N \lambda_k e_k  \| ^2 =  \langle x - \sum_{k=1}^N \lambda_k e_k, x - \sum_{j=1}^N \lambda_j e_j \rangle  = \]

    \[ =  \langle x, x \rangle  - 2 \langle x, \sum_{k=1}^N\lambda_ke_K \rangle  +  \langle \sum_{k=1}^N \lambda_k e_k, \sum_{j=1}^N \lambda_j e_j = \]

    \[ =  \| x \| ^2 - 2\sum_{k=1}^N \lambda_k c_k + \sum_{k, j=1}^N\lambda_k \lambda_j \lambda e_k, e_j \rangle = \]

    \[ =  \| x \| ^2 - 2 \sum_{k=1}^N\lambda_k c_k + \sum_{k=1}^N \lambda_k^2 \cdot 1 \]
    \[ =  \| x \| ^2 - \sum_{k=1}^N c_k^2 + (\sum_{k=1}^Nc_k^2 - 2\sum_{k=1}^N\lambda_k c_k + \sum_{k=1}^N \lambda_k^2) = \] 
    \[ =  \| x \| ^2 - \sum_{k=1}^N(\lambda_k - c_k)^2 = \]
    
    \[ =  \| x - \sum_{k=1}^N \lambda_k e_k \| ^2 \]

    Минимум: $\lambda_j = c_j, \qquad 1 \le j \le N$, т.е.

    \[ dist^2(x, span\{e_k\}_{K=1}^N) =  \| x \| ^2 - \sum_{k=1}^Nc_k^2 \ge 0 \ \ \forall \ \ N \]
    \[ \sum_{k=1}^N c_k^2 \le  \| x \| ^2 \qquad \forall N \ \ \Rightarrow \sum_{k=1}^\infty c_k^2 \] ~--- сходится и по неравенству Бесселя:

    \[ \sum_{k=1}^\infty c_k^2 \le  \| x \| ^2 \]

    Пусть теперь $\{ e_k\}_{k \in \NN}$~--- ортонормированный базис, т.е. $clos \ \ span \{ e_k \} = \mathcal{H} \Leftrightarrow$

    \[ \forall \ \ x \in \mathcal{H} \qquad \exists \ \ \{\lambda_k\} \st \sum_{k=1}^N \lambda_k e_k \rightarrow x (N \rightarrow \infty) \]

    т.е.:

    \[  \|  x - \sum_{k1}^N \lambda_k e_K \| ^2 \longrightarrow 0 \]

    \[  \| x \| ^2\cdot \sum_{k=1}^N c_k^2 \longrightarrow 0 (n \rightarrow \infty) \]
\end{proof}

\begin{Proposition}{}{}
    Если $\{ e_k\}_{k \in \NN}$~--- ортонормированный базис, то верно равенство Парсеваля:

    \[ \sum_{k=1}^\infty c_K^2 =  \| x \| ^2, \qquad c_k =  \langle x_k, x_k \rangle  \]
\end{Proposition}


Итак: $\ \ \mathcal{H}$~--- сепарабельное гильбертово пространство; $\{ e_k \}$~--- ортонормированный базис в $\mathcal{H}$

\[ \mathcal{H} \ni x \longrightarrow \{ c_k\}_{k=1}^\infty, \qquad c_k = \langle x, \rangle e_k\]


\begin{Theorem}{Рисс-Фишер}{}
    Пусть $\{ e_k\}$~--- ортонормированный базис в $\mathcal{H} \st c = (c_1, \ldots, c_k, \ldots) \in l^2 \Rightarrow$

    $\exists ! \ \ x \in \mathcal{H} \st c_k =  \langle x_1, e_k \rangle _\mathcal{H}$ и

    $ \| c \| _{l^2} =  \| x \| _{\mathcal{H}}$
\end{Theorem}

\begin{proof}

    Конечно же $x = \sum_{k=1}^\infty c_k e_k$

    Проверим, что ряд сходится, т.е. последовательность частичных сумм куда-то сходятся

    Проверим, что это последовательность Коши, т.е. 
    
    \[ \forall \ \ \eps  >  0 \ \ \exists \ \ N(c) \st \forall \ \ n \ge N(\eps), \forall \ \ n > N(\eps), m \in \NN: \]

    \[  \| S_{N+m} - S_N \|  < \eps \]

    \[  \| S_{N + m} - S_N \| ^2 =  \| \sum_{k = N+1}^{N+m} c_k e_k \|  \]

    но $c = (c_1, \ldots, c_k, \ldots) \in l^2, \qquad  \| C \| ^2 < +\infty = \sum_{k=1}^\infty c_k^2$

    \[ \forall \ \ \eps > 0 \ \ \exists \ \ N(\eps) \st \forall \ \ m \in \NN \st N \ge N(\eps), \ \ \text{верно} \ \ \sum_{k=N+1}^{N+m}c_k^2 < \eps \]

    т.е. ряд $\sum_{k=1}^\infty c_k e_k$ сходится, обозначим эту сумму за $x$


    Осталось проверить, что $ \langle x, e_k \rangle  = c_k$ и единственность

    \[  \langle x, e_k \rangle  \overset{def}{=}  \langle x - S_N, e_k \rangle _{\mathcal{H}} +  \langle S_N, e_k \rangle _{\mathcal{H}} \qquad \forall \ \ N \]

    \[  \langle S_N, e_k \rangle  =  \langle \sum_{j=1}^N c_je_j, e_k \rangle  = c_k \]

    \[ | \langle x - S_N, e_k \rangle | \le  \| f - S_N \|  \cdot  \| e_k \|  \longrightarrow 0 (N \rightarrow +\infty) \]

    Единственность:

    Пусть $ \langle x, y \rangle  =  \langle y, e_k \rangle  = c_k \qquad \forall \ \ k \Rightarrow$

    $(x - y) \perp e_k \ \ \forall \ \ k \Rightarrow (x - y) = 0$ т.к. $\{e_k\}$~--- ортонормированный базис

    Пусть $(x-y) = z \not = 0 \Rightarrow c'_k :=  \langle z, e_k \rangle  \not = 0$ (не равняется тождественно)

    \[ \text{dist}^2(z, \text{span} \{e_k \}_{k=1}^N) = \| z \| ^2 - \sum_{k=1}^N{c`_k^2} \]

    С другой стороны, $ \| z - \sum_{k=1}^N c'_k e_k \|  \longrightarrow 0 (N \rightarrow 0) (?!)$
\end{proof}

\textbf{Следствие:}

    Любое Гильбертово пространство $\mathcal{H} \ \ $ изоморфно $l^2$, т.е. есть биекция $\mathcal{A} \st \mathcal{H} \rightarrow e^2$:

    \[  \| \mathcal{A}x \| _{l^2} =  \| x \| _{l^2} \]

    $\mathcal{A}$~--- линейный оператор:

    $ \langle \mathcal{A}x, \mathcal{A}y \rangle _{l^2} =  \langle x, y \rangle _{\mathcal{H}}$

\begin{proof}

    Проверим линейность оператора $\mathcal{A}$:

    В $\mathcal{H}$~ есть ортонормированный базис $\{ e_k\}_{k+1}^\infty \st \mathcal{A}x = c = \{ c_k \}$
    
    $ \langle \mathcal{A}x, \mathcal{A}y \rangle _{l^2} \overset{\text{def}}{=} \sum_{k=1}^\infty c_k d_k$

    \[ 
    S_N = \sum_{k=1}^N c_k e_k \qquad \tilde{S}_N = \sum_{k=1}^N d_k e_k
    \]

    \[  \|  \langle x - S_N + S_N, y - \tilde{S}_N + \tilde{S}_N \rangle  \|  \le  \| x - S_N \| \cdot  \| y - \tilde{S_N} \|  \longrightarrow 0 (n \rightarrow \infty) \]

    \[ | \langle x - S_N, \tilde{S}_N \rangle | \longrightarrow 0 (n \rightarrow \infty) \]

    \[ | \langle S_N, y - \tilde{S_N}| \longrightarrow 0 (n \rightarrow \infty) \]

    \[  \langle S_N, \tilde{S_N} \rangle  =  \langle \sum_{k=1}^Nc_n e_k, \sum_{j=1}^N d_j e_j \rangle  = \sum_{k=1}^N c_k d_k \]

    \[ | \langle x, y \rangle _{\mathcal{H}} - \sum_{k=1}^\infty c_k d_k| \longrightarrow 0 (N \rightarrow \infty) \]

    \[ \langle x, y \rangle  = \sum_{k=1}^\infty c_k d_k \]

\end{proof}

\begin{Remark}{}{}
    Единичный шар $\overline{B_d(0, 1)} = \{ x \in l^2 :  \| x \| _{l^2} \le 1 \}$~--- не компактен.
\end{Remark}

Recall that:

$X$~--- банахово пространство, если $X$~--- линейное нормированное пространство, полное относительно метрики, порождаемой своей нормой

$\sum_{k=1}^\infty \lambda_k x_k$~--- сходится в $X$, если $\ \ \exists \ \ x \in X \st  \|  x - \sum_{k=1}^N\lambda_k x_k \| _X \longrightarrow 0 (N \rightarrow \infty)$

\begin{Definition}{}{}
    Пусть $X, Y$~--- линейные векторные пространства.

    Отображение $\mathcal{A} \st X \rightarrow Y$~--- \textbf{линейный оператор}, если оно линейное:

    \[ \mathcal{A}(x + \lambda y) = \mathcal{A}x + \lambda \mathcal{A}y, \qquad \forall \ \ x, y \in X, \lambda \in \RR \]

    Если $Y = \RR$, то линейный оператор называется \textbf{линейным функционалом}.
\end{Definition}

\begin{Definition}{}{}
    Пусть $X, Y$~--- банаховы. $\qquad \mathcal{A} \st X \rightarrow Y$~--- линейный оператор

    \[ \mathcal{A} \st Dom \mathcal{A} \rightarrow Range \mathcal{A} \]
    где $Dom \mathcal{A}$~--- линейное пространство X, а $Range \mathcal{A}$~--- линейное подпространство $Y$, т.е.
    
    если $x, y \in Dom \mathcal{A} \Rightarrow x + \lambda y \in Dom \mathcal{A}, \ \ \forall \ \ \lambda \in \RR$. Аналогично для $Range \mathcal{A}$. 
\end{Definition}

\begin{Example}{}{}
    $X = C([0, 1])$

    \[
    \| f \| _X := sup_{x \in [0, 1]} |f(x)|
    \]

    \[
    \mathcal{A}f := f' \qquad \mathcal{A} = \{ f \in C([0, 1]) \st \exists \exists f(x) \ \ \forall \ \ x \in (0, 1] ) \}
    \]

\end{Example}

Зададим норму оператора $\mathcal{A}$:

\[  \| \mathcal{A} := \sup_{ \| x \|  \le 1}  \| \mathcal{A}x \| _Y \]

Оператор называется ограниченным, если $ \| \mathcal{A} \|   <  +\infty$

\begin{Example}{}{}
    $(x^k)' = kx^{k-1} = \RR$, при $x = 1$

    \[
    \| x^k \| _{C([0, 1])} = max_{0 \le x \le 1} |x^k| = 1
    \]

    Пример неограниченного оператора:

    $(\mathcal{A}f)(x) := f(x)g(x), \qquad \text{где} \ \ g \in C([0,1])$~--- фиксирована

    \[  \| \mathcal{A} \|  := \sup_{ \| f \| _{C([0,1])}}  \| \mathcal{A}f \|  = \sup_{x \in [0, 1], f \in C([0,1 ])} |f(x)g(x)| \le \]

    \[ \sup |f(x)| \cdot \sup |g(x)|  \langle  +\infty \]
\end{Example}

\begin{Proposition}{}{}
    Пусть $\mathcal{A} \st X \rightarrow Y$~--- линейный, $\qquad X, Y$~--- банаховы пространства, тогда

    \[  \| \mathcal{A} \|  = \sup_{ \| x \| _X = 1} \| \mathcal{A}x \| _Y = \sup_{x \in X, x \not = 0} \frac{ \| \mathcal{A}x \| }{ \| x \| } \]
\end{Proposition}

\begin{proof}

    \[  \| \mathcal{A} \|  \ge \sup_{ \| x \| _X = 1}  \| \mathcal{A}x \| _Y \qquad (\text{на самом деле равенство})\]

    \[ \forall x \not = 0 \st  \| \mathcal{A}x \| _Y =  \| \mathcal{A}(\frac{x}{ \| x \| _X}) \| _Y\cdot  \| x \| _X \]

    \[ x = 0 \Rightarrow  \| \mathcal{A}x \| _Y = 0 \]
\end{proof}

\begin{Proposition}{}{}

    $X, Y$~--- банаховы пространства.
    
    \[
    \mathcal{A} \st X \rightarrow Y \text{~--- ограничен} \Leftrightarrow \mathcal{A} \text{~--- непрерывный как отображение} \Leftrightarrow \text{непрерывный в нуле.}
    \]

\end{Proposition}

\begin{proof}

    \begin{enumerate}
        \item Непрерывность в нуле $\Rightarrow$ непрерывность
        
        Неравенство в нуле:
        \[ \forall \ \ \eps  >  0 \ \ \exists \ \ \delta  >  0 \qquad  \| x \| _X  <  \delta \Rightarrow  \| \mathcal{A}x \| _Y  <  \eps \]

        \item Непрерывность в точке $Y \in Dom \mathcal{A}$
        
        \[ \forall \ \ \eps  >  0 \ \ \exists \ \ \delta  >  0 \qquad  \| y-x \| _X  <  \delta \Rightarrow  \| \mathcal{A}y - \mathcal{A}x \|  =  \| \mathcal{A}(y-x) \|   <  \eps \]

        \item Непрерывность $\Rightarrow$ ограниченность
        
        Покажем, что если $\mathcal{A}$~--- непрерывный, то $ \| \mathcal{A} \|   <  + \infty$

        Рассмотрим $\sup_{ \| x \| _X \le 1}  \| \mathcal{A}x \| _Y$

        Непрерывность в нуле: 

        \[ \forall \ \ \eps  >  0 \ \ \exists \ \ \delta  >  0 \st  \| x \| _X \le \delta \Rightarrow  \| \mathcal{A}x \| _Y \le \eps \]

        \[ \eps := 1 \qquad  \| \mathcal{A}x \|  = \frac{1}{\delta} \| \mathcal{A}(\frac{1}{\delta}) \| \]

        \[ \forall \ \ x' \st  \| x' \| _X \le 1  \| \delta x' \| _x \le \delta \Rightarrow  \| \mathcal{A}x' \| _Y = \frac{1}{\delta} \| \mathcal{A}(x'\delta) \| _Y \le \frac{1}{\delta} \]

        \[ \sup_{ \| x' \| _x \le 1}  \| \mathcal{A}x' \| _Y \le \frac{1}{\delta} \Rightarrow  \| \mathcal{A} \|  \le \frac{1}{\delta} \]

        \item Ограниченность $\Rightarrow$ непрерывность в нуле
        
        \[  \| \mathcal{A} x \|_Y \le  \|\mathcal{A} \| \cdot \| x \|_X \ \ \forall \ \ x \in X \]
        \[ \| \mathcal{A} \| = \sup_{X \ni x \not = 0} \frac{\|\mathcal{A}x\|}{\| x \|} \]
    \end{enumerate}
\end{proof}
